45/**
46 * @ingroup debug-internal
47 * @brief Implementation class for runtime assertions with formatted messages and diagnostics.
48 *
49 * This is the internal implementation class that powers the `debug::Assert` public API.
50 * Users should not instantiate this class directly; instead, use the `debug::Assert`
51 * template alias.
52 *
53 * In debug builds (NDEBUG not defined), the constructor:
54 * 1. Evaluates the predicate condition
55 * 2. If false, prints formatted assertion message, source location, and stack trace
56 * 3. Calls `BreakpointIfDebugging()` to break if a debugger is attached
57 * 4. Calls `std::abort()` to terminate the program
58 *
59 * The implementation supports multiple build configurations:
60 * - **Standard library** (`__LOGENIUM_DEBUG_USE_FMTLIB__ == 0`): Uses `std::format` and `std::println`
61 * - **fmtlib** (`__LOGENIUM_DEBUG_USE_FMTLIB__ == 1`): Uses `fmt::format` and `fmt::println`
62 * - **Colored output** (`__LOGENIUM_DEBUG_USE_COLOR_LOGS__ == 1`, requires fmtlib): Highlights
63 * "Assertion failed" with red background and white text
64 * - **Fast stacktrace** (`__LOGENIUM_DEBUG_USE_FAST_STACKTRACE__ == 1`): Prints stacktrace
65 * as-is instead of in reverse order
66 *
67 * In release builds (NDEBUG defined), the constructor is constexpr and does nothing,
68 * allowing the compiler to completely eliminate assertion checks and formatting overhead.
69 *
70 * @tparam Args Variadic template parameters for format arguments
71 *
72 * @note This is an implementation detail. Use `debug::Assert` instead.
73 *
74 * @see Assert for the public API
75 */
76template <typename... Args>
77struct AssertImpl {
78#ifndef NDEBUG
79
80 /**
81 * @brief Constructor for debug builds using fmtlib formatting.
82 *
83 * Evaluates the predicate and, if false, prints diagnostic information and aborts.
84 * This version uses fmtlib for formatting and optionally colored output.
85 *
86 * @param predicate The condition to check
87 * @param format Format string (fmt::format_string)
88 * @param args Arguments for the format string
89 * @param location Source location (automatically captured)
90 * @param stacktrace Stack trace (automatically captured)
91 *
92 * Diagnostic output includes:
93 * - Formatted assertion message (with red background if __LOGENIUM_DEBUG_USE_COLOR_LOGS__)
94 * - Source file, line, and function name
95 * - Full stack trace (reversed for readability, unless __LOGENIUM_DEBUG_USE_FAST_STACKTRACE__)
96 *
97 * If predicate is false:
98 * 1. Prints diagnostics to stderr (with color if enabled)
99 * 2. Calls IsDebuggerPresent()
100 * 3. Calls Breakpoint() if a debugger is attached (otherwise calls std::abort())
101 */
102 AssertImpl(bool predicate, fmt::format_string<Args...> format, Args &&...args,
103 std::source_location location = std::source_location::current(),
104 const std::stacktrace &stacktrace = std::stacktrace::current()) {
105 ZoneScoped;
106 if (!predicate) {
107 ZoneScopedN("Assertion handler");
108
109 fmt::println("{}: {}",
110 fmt::styled("Assertion failed", fmt::bg(fmt::color::red) | fmt::fg(fmt::color::white)),
111 fmt::format(format, std::forward<Args>(args)...));
112
113 fmt::println("Location: {}:{} in {}", location.file_name(), location.line(), location.function_name());
114
115#if __LOGENIUM_DEBUG_USE_FAST_STACKTRACE__
116
117 fmt::println("Stacktrace:\n{}", std::to_string(stacktrace));
118
119#else
120
121 fmt::println("Stacktrace:");
122 auto frame_count = std::distance(stacktrace.crbegin(), stacktrace.crend());
123 for (auto frame = stacktrace.crbegin(); frame != stacktrace.crend(); ++frame) {
124 fmt::println(" {}: {}", frame_count--, std::to_string(*frame));
125 }
126
127#endif // __LOGENIUM_DEBUG_USE_FAST_STACKTRACE__
128
129 if (IsDebuggerPresent()) {
130 Breakpoint();
131 } else {
132 std::abort();
133 }
134 }
135 }
136
137#else
138
139 /**
140 * @brief Constexpr no-op constructor for release builds (fmt::format version).
141 *
142 * In release builds, this constructor is constexpr and does nothing. The compiler
143 * completely eliminates assertion checks and all formatting overhead.
144 *
145 * @param predicate Ignored
146 * @param format Ignored
147 * @param args Ignored
148 */
149 constexpr AssertImpl(bool predicate, fmt::format_string<Args...> format, Args &&...args) {}
150
151#endif // NDEBUG
152};

156 * @brief Deduction guide for fmt::format_string (fmtlib version).
157 */

159AssertImpl(bool, fmt::format_string<Args...>, Args &&...) -> AssertImpl<Args...>;

163/**
164 * @ingroup debug
165 * @brief Runtime assertion with formatted messages, source location, and stack traces.
166 *
167 * Assert provides enhanced runtime checking with detailed diagnostic output. Unlike
168 * standard `<cassert>`, it supports formatted messages, automatically captures source
169 * location and stack traces, and can optionally break into the debugger before aborting.
170 *
171 * ## Usage
172 *
173 * Assertions are created using brace initialization syntax:
174 * @code
175 * debug::Assert{condition, "format string", args...};
176 * @endcode
177 *
178 * The assertion checks `condition` and, if false:
179 * 1. Prints the formatted message with source location and stack trace
180 * 2. Calls `BreakpointIfDebugging()` to break if a debugger is attached
181 * 3. Calls `std::abort()` to terminate the program
182 *
183 * ## Build Configurations
184 *
185 * The implementation adapts to different build configurations:
186 *
187 * ### Debug vs Release
188 * - **Debug builds** (`NDEBUG` not defined): Full assertion checking with diagnostics
189 * - **Release builds** (`NDEBUG` defined): Constexpr no-op, completely optimized away
190 *
191 * ### Formatting Library
192 * - **Standard library** (default): Uses `std::format` and `std::println`
193 * - **fmtlib** (`LOGENIUM_DEBUG_USE_FMTLIB=ON`): Uses `fmt::format` and `fmt::println`
194 *
195 * ### Colored Output
196 * - **Disabled** (default): Plain text output
197 * - **Enabled** (`LOGENIUM_DEBUG_USE_COLOR_LOGS=ON`, requires fmtlib): "Assertion failed"
198 * highlighted with red background and white text
199 *
200 * ### Stack Trace Format
201 * - **Reversed** (default): Stack frames printed in reverse order (most recent last)
202 * - **Fast** (`LOGENIUM_DEBUG_USE_FAST_STACKTRACE=ON`): Stack frames in natural order
203 *
204 * ## Examples
205 *
206 * @code
207 * // Simple assertion with message
208 * void process(int *ptr) {
209 * debug::Assert{ptr != nullptr, "Null pointer passed to process()"};
210 * // ... use ptr safely ...
211 * }
212 * @endcode
213 *
214 * @code
215 * // Assertion with formatted message
216 * void set_value(int value) {
217 * debug::Assert{value >= 0 && value <= 100,
218 * "Value {} out of range [0, 100]", value};
219 * // ... proceed with valid value ...
220 * }
221 * @endcode
222 *
223 * @code
224 * // Complex condition with detailed diagnostics
225 * void transfer(Account &from, Account &to, double amount) {
226 * debug::Assert{from.balance >= amount,
227 * "Insufficient funds: account {} has ${:.2f}, need ${:.2f}",
228 * from.id, from.balance, amount};
229 * // ... perform transfer ...
230 * }
231 * @endcode
232 *
233 * @code
234 * // Mathematical invariant checking
235 * int divide(int a, int b) {
236 * debug::Assert{b != 0, "Division by zero: {} / {}", a, b};
237 * return a / b;
238 * }
239 * @endcode
240 *
241 * ## Why Use Assert Over <cassert>?
242 *
243 * - **Formatted messages**: Use modern formatting instead of plain strings
244 * - **Automatic context**: Source location captured without manual `__FILE__`/`__LINE__`
245 * - **Stack traces**: See the full call chain leading to the failure
246 * - **Debugger integration**: Automatically breaks if a debugger is attached
247 * - **Better diagnostics**: Clear, formatted output helps debug issues faster
248 *
249 * @tparam Args Variadic template parameters for format arguments (deduced automatically)
250 *
251 * @note Always prefer `debug::Assert` over `<cassert>` in this codebase for better
252 * diagnostics and debugger integration.
253 *
254 * @note In release builds, assertions compile away completely with zero overhead.
255 *
256 * @note Source location and stack traces are captured automatically at the assertion
257 * site, not at the AssertImpl constructor site.
258 *
259 * @see BreakpointIfDebugging() for the automatic debugger breaking behavior
260 * @see detail::AssertImpl for implementation details
261 */
262template <typename... Args>
263using Assert = detail::AssertImpl<Args...>;
