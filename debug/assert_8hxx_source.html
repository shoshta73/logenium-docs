52/**
53 * @ingroup debug-internal
54 * @brief Implementation class for runtime assertions with formatted messages and diagnostics.
55 *
56 * This is the internal implementation class that powers the `debug::Assert` public API.
57 * Users should not instantiate this class directly; instead, use the `debug::Assert`
58 * template alias.
59 *
60 * In debug builds (NDEBUG not defined), the constructor:
61 * 1. Evaluates the predicate condition
62 * 2. If false, prints formatted assertion message, source location, and stack trace
63 * 3. Calls `BreakpointIfDebugging()` to break if a debugger is attached
64 * 4. Calls `std::abort()` to terminate the program
65 *
66 * The implementation supports multiple build configurations:
67 * - **Standard library** (`__LOGENIUM_DEBUG_USE_FMTLIB__ == 0`): Uses `std::format` and `std::println`
68 * - **fmtlib** (`__LOGENIUM_DEBUG_USE_FMTLIB__ == 1`): Uses `fmt::format` and `fmt::println`
69 * - **Colored output** (`__LOGENIUM_DEBUG_USE_COLOR_LOGS__ == 1`, requires fmtlib): Highlights
70 * "Assertion failed" with red background and white text
71 * - **Fast stacktrace** (`__LOGENIUM_DEBUG_USE_FAST_STACKTRACE__ == 1`): Prints stacktrace
72 * as-is instead of in reverse order
73 *
74 * In release builds (NDEBUG defined), the constructor is constexpr and does nothing,
75 * allowing the compiler to completely eliminate assertion checks and formatting overhead.
76 *
77 * @tparam Args Variadic template parameters for format arguments
78 *
79 * @note This is an implementation detail. Use `debug::Assert` instead.
80 *
81 * @see Assert for the public API
82 */
83template <typename... Args>
84struct AssertImpl {
85#ifndef NDEBUG
86
87#if !__LOGENIUM_DEBUG_USE_FMTLIB__

89 /**
90 * @brief Constructor for debug builds using standard library formatting.
91 *
92 * Evaluates the predicate and, if false, prints diagnostic information and aborts.
93 *
94 * @param predicate The condition to check
95 * @param format Format string (std::format_string)
96 * @param args Arguments for the format string
97 * @param location Source location (automatically captured)
98 * @param stacktrace Stack trace (automatically captured)
99 *
100 * Diagnostic output includes:
101 * - Formatted assertion message
102 * - Source file, line, and function name
103 * - Full stack trace (reversed for readability, unless __LOGENIUM_DEBUG_USE_FAST_STACKTRACE__)
104 *
105 * If predicate is false:
106 * 1. Prints diagnostics to stderr
107 * 2. Calls IsDebuggerPresent()
108 * 3. Calls Breakpoint() if a debugger is attached (otherwise calls std::abort())
109 */
110 AssertImpl(bool predicate, std::format_string<Args...> format, Args &&...args,
111 std::source_location location = std::source_location::current(),
112 const std::stacktrace &stacktrace = std::stacktrace::current()) {

114 std::println("Assertion failed: {}", std::format(format, std::forward<Args>(args)...));

117#if __LOGENIUM_DEBUG_USE_FAST_STACKTRACE__

119 std::println("Stacktrace:\n{}", std::to_string(stacktrace));
120
121#else
122
123 std::println("Stacktrace:");
124 for (auto frame = stacktrace.crbegin(); frame != stacktrace.crend(); ++frame) {
125 std::println(" {}", std::to_string(*frame));

128#endif // __LOGENIUM_DEBUG_USE_FAST_STACKTRACE__

261/**
262 * @ingroup debug
263 * @brief Runtime assertion with formatted messages, source location, and stack traces.
264 *
265 * Assert provides enhanced runtime checking with detailed diagnostic output. Unlike
266 * standard `<cassert>`, it supports formatted messages, automatically captures source
267 * location and stack traces, and can optionally break into the debugger before aborting.
268 *
269 * ## Usage
270 *
271 * Assertions are created using brace initialization syntax:
272 * @code
273 * debug::Assert{condition, "format string", args...};
274 * @endcode
275 *
276 * The assertion checks `condition` and, if false:
277 * 1. Prints the formatted message with source location and stack trace
278 * 2. Calls `BreakpointIfDebugging()` to break if a debugger is attached
279 * 3. Calls `std::abort()` to terminate the program
280 *
281 * ## Build Configurations
282 *
283 * The implementation adapts to different build configurations:
284 *
285 * ### Debug vs Release
286 * - **Debug builds** (`NDEBUG` not defined): Full assertion checking with diagnostics
287 * - **Release builds** (`NDEBUG` defined): Constexpr no-op, completely optimized away
288 *
289 * ### Formatting Library
290 * - **Standard library** (default): Uses `std::format` and `std::println`
291 * - **fmtlib** (`LOGENIUM_DEBUG_USE_FMTLIB=ON`): Uses `fmt::format` and `fmt::println`
292 *
293 * ### Colored Output
294 * - **Disabled** (default): Plain text output
295 * - **Enabled** (`LOGENIUM_DEBUG_USE_COLOR_LOGS=ON`, requires fmtlib): "Assertion failed"
296 * highlighted with red background and white text
297 *
298 * ### Stack Trace Format
299 * - **Reversed** (default): Stack frames printed in reverse order (most recent last)
300 * - **Fast** (`LOGENIUM_DEBUG_USE_FAST_STACKTRACE=ON`): Stack frames in natural order
301 *
302 * ## Examples
303 *
304 * @code
305 * // Simple assertion with message
306 * void process(int *ptr) {
307 * debug::Assert{ptr != nullptr, "Null pointer passed to process()"};
308 * // ... use ptr safely ...
309 * }
310 * @endcode
311 *
312 * @code
313 * // Assertion with formatted message
314 * void set_value(int value) {
315 * debug::Assert{value >= 0 && value <= 100,
316 * "Value {} out of range [0, 100]", value};
317 * // ... proceed with valid value ...
318 * }
319 * @endcode
320 *
321 * @code
322 * // Complex condition with detailed diagnostics
323 * void transfer(Account &from, Account &to, double amount) {
324 * debug::Assert{from.balance >= amount,
325 * "Insufficient funds: account {} has ${:.2f}, need ${:.2f}",
326 * from.id, from.balance, amount};
327 * // ... perform transfer ...
328 * }
329 * @endcode
330 *
331 * @code
332 * // Mathematical invariant checking
333 * int divide(int a, int b) {
334 * debug::Assert{b != 0, "Division by zero: {} / {}", a, b};
335 * return a / b;
336 * }
337 * @endcode
338 *
339 * ## Why Use Assert Over <cassert>?
340 *
341 * - **Formatted messages**: Use modern formatting instead of plain strings
342 * - **Automatic context**: Source location captured without manual `__FILE__`/`__LINE__`
343 * - **Stack traces**: See the full call chain leading to the failure
344 * - **Debugger integration**: Automatically breaks if a debugger is attached
345 * - **Better diagnostics**: Clear, formatted output helps debug issues faster
346 *
347 * @tparam Args Variadic template parameters for format arguments (deduced automatically)
348 *
349 * @note Always prefer `debug::Assert` over `<cassert>` in this codebase for better
350 * diagnostics and debugger integration.
351 *
352 * @note In release builds, assertions compile away completely with zero overhead.
353 *
354 * @note Source location and stack traces are captured automatically at the assertion
355 * site, not at the AssertImpl constructor site.
356 *
357 * @see BreakpointIfDebugging() for the automatic debugger breaking behavior
358 * @see detail::AssertImpl for implementation details
359 */
360template <typename... Args>
361using Assert = detail::AssertImpl<Args...>;
