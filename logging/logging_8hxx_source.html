8 * @file logging.hxx
9 * @brief Header-only logging utility with compile-time format string validation
10 *
11 * This header provides a simple, type-safe logging interface with compile-time
12 * format string validation. The logging backend can use either C++20 standard
13 * library formatting (std::format/std::println) or fmtlib (fmt::format/fmt::println)
14 * based on the LOGENIUM_LOGGING_USE_FMTLIB CMake option.
15 */
16
17/**
18 * @defgroup logging Logging Module
19 * @brief Simple, type-safe logging with compile-time format string validation
20 *
21 * The logging module provides a header-only logging utility that wraps format
22 * string validation and output to stdout. It supports two backends:
23 *
24 * - **Standard library** (default when LOGENIUM_LOGGING_USE_FMTLIB=OFF):
25 * Uses std::format_string, std::println from C++20 <format> and <print>
26 *
27 * - **fmtlib** (when LOGENIUM_LOGGING_USE_FMTLIB=ON):
28 * Uses fmt::format_string, fmt::println from fmtlib v12.1.0
29 *
30 * ## Features
31 *
32 * - **Compile-time validation**: Format strings are checked at compile time
33 * - **Type safety**: Arguments are type-checked against the format string
34 * - **Source location tracking**: Automatically captures and displays file, line, and function
35 * - **Level-based logging**: Supports 10 severity levels from Trace3 to Fatal
36 * - **Perfect forwarding**: Arguments are forwarded without unnecessary copies
37 * - **Zero overhead**: Template-based design with no runtime cost
38 *
39 * ## Usage
40 *
41 * Basic logging with automatic type deduction and level selection:
42 * @code
43 * #include <logging/logging.hxx>
44 *
45 * // Default Info level
46 * logging::log("Hello, {}!", "world");
47 * // Output: [Info] Hello, world! (file.cxx:42 in main)
48 *
49 * // Level-specific logging
50 * logging::debug("Debug value: {}", value);
51 * logging::warn("Warning: {}", condition);
52 * logging::error("Error occurred: {}", error_msg);
53 * logging::fatal("Fatal error: {}", critical_error);
54 *
55 * // All levels: trace3, trace2, trace1, trace, debug, info, warn, error, fatal
56 * @endcode
57 *
58 * ## Output Format
59 *
60 * All log messages follow this format:
61 * @code
62 * [Level] message (file:line in function)
63 * @endcode
64 *
65 * ## Implementation Notes
66 *
67 * The logging module is header-only and uses template metaprogramming to provide
68 * compile-time format string validation. The actual formatting backend is selected
69 * at compile time based on the __LOGENIUM_LOGGING_USE_FMTLIB__ preprocessor macro.
70 *
71 * Source location is automatically captured via std::source_location::current()
72 * and displayed in every log message, providing precise debugging information.
73 */

101/**
102 * @ingroup logging_internal
103 * @brief Implementation struct for compile-time validated logging
104 *
105 * This struct provides the actual logging implementation with compile-time
106 * format string validation. It uses CTAD (Class Template Argument Deduction)
107 * to automatically deduce template arguments from constructor parameters.
108 *
109 * @tparam L The logging level for this log message
110 * @tparam Args Variadic template parameter pack for format arguments
111 *
112 * @note This is an implementation detail. Users should use the logging::log
113 * family of alias templates instead of instantiating LogImpl directly.
114 */
115template <Level L, typename... Args>
116struct LogImpl {
117 /**
118 * @brief Construct and execute a log operation using fmtlib
119 *
120 * Validates the format string at compile time and outputs the formatted
121 * message to stdout using fmt::println with the format:
122 * `[Level] message (file:line in function)`
123 *
124 * @param format Format string with compile-time validation (fmt::format_string)
125 * @param args Variadic arguments to be formatted (forwarded)
126 * @param location Source location for debugging (automatically captured)
127 *
128 * @note Arguments are perfectly forwarded to avoid unnecessary copies
129 * @note Source location is automatically captured and displayed in output
130 */
131 LogImpl(fmt::format_string<Args...> format, Args &&...args,
132 std::source_location location = std::source_location::current()) {
133 ZoneScoped;
134
135 auto level_string = fmt::format(internal::utils::ToColor(L), "{}", L);
136
137 fmt::println("[{}] {} ({}:{} in {})", level_string, fmt::format(format, std::forward<Args>(args)...),
138 location.file_name(), location.line(), location.function_name());
139 }
140};
141
142/**
143 * @ingroup logging_internal
144 * @brief Deduction guide for LogImpl with fmtlib backend
145 *
146 * Enables CTAD (Class Template Argument Deduction) to automatically deduce
147 * template parameters from constructor arguments when using fmtlib.
148 *
149 * @tparam Args Deduced argument types from constructor
150 */
151template <Level L = Level::Ignore, typename... Args>
152LogImpl(fmt::format_string<Args...>, Args &&...) -> LogImpl<L, Args...>;
153
154} // namespace detail
