11/**
12 * @ingroup type_traits
13 * @brief Type trait to create const pointer or const reference based on input type.
14 *
15 * This trait produces different const-qualified types depending on whether the
16 * input is a pointer or not:
17 * - For pointer types: creates a const pointer (pointer itself is const)
18 * - For non-pointer types: creates a const lvalue reference
19 *
20 * Behavior:
21 * - For pointer types: `T*` → `T* const` (const pointer to T)
22 * - For non-pointer types: `T` → `const T&` (const reference to T)
23 *
24 * This is useful for creating const-correct parameter types where pointers should
25 * be const pointers and non-pointers should be const references.
26 *
27 * @tparam T The type to transform into const pointer or const reference
28 *
29 * ## Member Types
30 *
31 * - `type`: The resulting const pointer or const reference type
32 *
33 * ## Helper Type Alias
34 *
35 * `const_pointer_or_const_ref_t<T>` provides a convenient way to access the type:
36 * @code
37 * using T = const_pointer_or_const_ref_t<int>; // const int&
38 * @endcode
39 *
40 * @note This trait is primarily used internally by the casting module for creating
41 * const-correct types in return type calculations
42 * @note For pointer types, the pointer itself becomes const, not the pointee
43 *
44 * Example:
45 * @code
46 * // Non-pointer types - creates const reference
47 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<int>, const int&>);
48 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<double>, const double&>);
49 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<std::string>, const std::string&>);
50 *
51 * // Pointer types - creates const pointer (pointer itself is const)
52 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<int*>, int* const>);
53 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<char*>, char* const>);
54 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<const int*>, const int* const>);
55 *
56 * // Note: For pointers, the pointer is const, not the pointee
57 * int* const p1 = ...; // const pointer to int (can't change pointer)
58 * const int* p2 = ...; // pointer to const int (can't change pointee)
59 *
60 * // Already const types
61 * static_assert(std::is_same_v<const_pointer_or_const_ref_t<const int>, const int&>);
62 *
63 * // Use case: const-correct function parameters
64 * template <typename T>
65 * void process(const_pointer_or_const_ref_t<T> param) {
66 * // param is const int& for int, int* const for int*, etc.
67 * }
68 * @endcode
69 */
70template <typename T>
71struct const_pointer_or_const_ref {
72 using type = std::conditional_t<std::is_pointer_v<T>, const T, const T &>;
73};
74
75/**
76 * @ingroup type_traits
77 * @brief Helper type alias for const_pointer_or_const_ref.
78 *
79 * Provides convenient access to const_pointer_or_const_ref<T>::type following
80 * the C++14 standard library convention.
81 *
82 * @tparam T The type to transform into const pointer or const reference
83 *
84 * Example:
85 * @code
86 * template <typename T>
87 * void safe_read(const_pointer_or_const_ref_t<T> value) {
88 * // value is a const reference for non-pointers, const pointer for pointers
89 * // Ensures the function can't modify the original value
90 * }
91 * @endcode
92 */
93template <typename T>
94using const_pointer_or_const_ref_t = typename const_pointer_or_const_ref<T>::type;
