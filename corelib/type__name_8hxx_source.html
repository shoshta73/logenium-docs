12/**
13 * @ingroup utility
14 * @brief Extract the type name of a template parameter at compile time.
15 *
16 * This function uses compiler intrinsics (__PRETTY_FUNCTION__ for GCC/Clang,
17 * __FUNCSIG__ for MSVC) to extract the type name as a string view at compile time.
18 *
19 * @tparam T The type to extract the name from
20 * @return constexpr std::string_view The type name as a string view
21 *
22 * @note This function is constexpr and can be evaluated at compile time
23 * @note Requires Clang, GCC, or MSVC compiler
24 *
25 * Example:
26 * @code
27 * constexpr auto name = type_name<int>(); // Returns "int"
28 * constexpr auto name2 = type_name<const std::string>(); // Returns "const std::basic_string<char>"
29 * @endcode
30 */
31template <typename T>
32constexpr std::string_view type_name() {
33#if defined(__clang__) || defined(__GNUC__)
34
35 constexpr auto prefix = std::string_view{"="};
36 constexpr auto suffix = std::string_view{"]"};
37 constexpr auto function = std::string_view{__PRETTY_FUNCTION__};
38
39 constexpr auto first = function.find_first_not_of(' ', function.find_first_of(prefix) + 1);
40 constexpr auto value = function.substr(first, function.find_last_of(suffix) - first);
41 return value;
42
43#elif defined(_MSC_VER)
44
45 constexpr auto function = std::string_view{__FUNCSIG__};
46 constexpr auto prefix = std::string_view{"type_name<"};
47 constexpr auto suffix = std::string_view{">(void)"};
48
49 constexpr auto start = function.find(prefix) + prefix.size();
50 constexpr auto end = function.rfind(suffix);
51
52 constexpr auto value = function.substr(start, end - start);
53 return value;
54
55#else
56#error "corelib/type_name.hxx: Unsupported compiler"
57#endif
58}

60/**
61 * @ingroup utility
62 * @brief Extract the type name from a value at compile time.
63 *
64 * This overload takes a forwarding reference to preserve cv-qualifiers and
65 * extracts the type name by removing the reference and calling the template
66 * version.
67 *
68 * @tparam T The deduced type from the value (automatically deduced)
69 * @param value The value to extract the type name from (not evaluated)
70 * @return constexpr std::string_view The type name as a string view
71 *
72 * @note This function is constexpr and can be evaluated at compile time
73 * @note The value parameter is not evaluated, only used for type deduction
74 * @note Reference qualifiers are stripped, but cv-qualifiers are preserved
75 *
76 * Example:
77 * @code
78 * int x = 42;
79 * const int y = 10;
80 * auto name1 = type_name(x); // Returns "int"
81 * auto name2 = type_name(y); // Returns "int" (const is part of type, not reference)
82 * auto name3 = type_name(std::move(x)); // Returns "int"
83 * @endcode
84 *
85 * @todo Implement not stripping reference qualifiers
86 */
87template <typename T>
88inline constexpr std::string_view type_name(T &&value) {
89 using Type = std::remove_reference_t<T>;
90 return type_name<Type>();
91}
