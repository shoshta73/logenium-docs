11/**
12 * @brief CRTP base class template for implementing extensible RTTI hierarchies.
13 * @ingroup extensible-rtti
14 *
15 * This template provides automatic implementation of the extensible RTTI interface
16 * for derived classes. It uses the Curiously Recurring Template Pattern (CRTP) to
17 * inject type-specific implementations of TypeID(), DynamicTypeID(), IsA(), and
18 * classof() methods.
19 *
20 * The template supports both single and multiple inheritance through variadic
21 * template parameters. The first parent type is the primary parent, and additional
22 * parents can be specified to enable multiple inheritance.
23 *
24 * @tparam ThisType The derived class type (CRTP parameter)
25 * @tparam ParentType The primary parent class type
26 * @tparam ParentTypes Additional parent types for multiple inheritance (optional)
27 *
28 * @note Requirements for ThisType:
29 * - Must declare a static member: `static char ID;`
30 * - Must define ID in implementation file: `char ThisType::ID = 0;`
31 * - Must implement: `void anchor() override {}`
32 * - Should declare Extends as friend: `friend class Extends<ThisType, ParentType, ...>;`
33 *
34 * @note This class automatically provides:
35 * - Static and dynamic type identification
36 * - Type checking via IsA() methods
37 * - Integration with LLVM-style casting (isa, cast, dyn_cast)
38 *
39 * Single Inheritance Example:
40 * @code
41 * class MyType : public Extends<MyType, Base> {
42 * public:
43 * MyType() {}
44 * private:
45 * static char ID;
46 * void anchor() override {}
47 * friend class Extends<MyType, Base>;
48 * };
49 *
50 * char MyType::ID = 0;
51 * @endcode
52 *
53 * Multiple Inheritance Example:
54 * @code
55 * class MyMultiType : public Extends<MyMultiType, ParentA, ParentB> {
56 * public:
57 * MyMultiType() {}
58 * private:
59 * static char ID;
60 * void anchor() override {}
61 * friend class Extends<MyMultiType, ParentA, ParentB>;
62 * };
63 *
64 * char MyMultiType::ID = 0;
65 * @endcode
66 *
67 * @see Base
68 */
69template <typename ThisType, typename ParentType, typename... ParentTypes>
70class Extends : public ParentType, public ParentTypes... {
71 public:
72 /**
73 * @brief Type alias for the primary parent type.
74 *
75 * @note This shadows the template parameter name but provides convenient
76 * access to the primary parent type within the derived class.
77 */
78 using ParentType = ParentType;
79
80 /**
81 * @brief Returns the static type identifier for ThisType.
82 *
83 * Returns the address of the static ID member in ThisType. This address
84 * serves as a unique compile-time constant identifier for the type.
85 *
86 * @return Pointer to ThisType::ID
87 */
88 static const void *TypeID() {
89 CRLB_ZONE_SCOPED;
90 return &ThisType::ID;
91 }
92
93 /**
94 * @brief Returns the dynamic type identifier for this object.
95 *
96 * Overrides the pure virtual DynamicTypeID() from Base to return the
97 * type identifier for ThisType. This enables runtime type identification.
98 *
99 * @return Pointer to ThisType::ID
100 */
101 [[nodiscard]] const void *DynamicTypeID() const override {
102 CRLB_ZONE_SCOPED;
103 return &ThisType::ID;
104 }
105
106 /**
107 * @brief Template-based type checking.
108 *
109 * Checks if this object is an instance of or derived from QueryType by
110 * comparing type identifiers. This is a convenience wrapper around the
111 * pointer-based IsA() method.
112 *
113 * @tparam QueryType The type to check against
114 * @return true if this object is of type QueryType or derives from it
115 *
116 * Example:
117 * @code
118 * MyType obj;
119 * if (obj.IsA<Base>()) {
120 * // obj derives from Base
121 * }
122 * if (obj.IsA<MyType>()) {
123 * // obj is exactly MyType
124 * }
125 * @endcode
126 */
127 template <typename QueryType>
128 [[nodiscard]] bool IsA() const {
129 CRLB_ZONE_SCOPED;
130 return IsA(QueryType::TypeID());
131 }
132
133 /**
134 * @brief Pointer-based type checking with inheritance chain traversal.
135 *
136 * Checks if this object is an instance of or derived from the type identified
137 * by TID. This method checks the entire inheritance hierarchy including all
138 * parent types.
139 *
140 * For multiple inheritance, this checks:
141 * 1. ThisType's TypeID (exact match)
142 * 2. Primary parent's inheritance chain
143 * 3. All additional parents' inheritance chains (if any)
144 *
145 * @param TID Type identifier to check against (from TypeID())
146 * @return true if this object is of type TID or derives from it
147 *
148 * @note The implementation uses fold expressions (C++17) to check all parent
149 * types in multiple inheritance scenarios.
150 */
151 bool IsA(const void *const TID) const override {
152 CRLB_ZONE_SCOPED;
153 return TID == TypeID() || ParentType::IsA(TID) || (ParentTypes::IsA(TID) || ...);
154 }
155
156 /**
157 * @brief LLVM-style classof() method for casting integration.
158 *
159 * Provides the classof pattern required by corelib's LLVM-style casting
160 * infrastructure (isa, cast, dyn_cast). This method enables type checking
161 * on pointer types.
162 *
163 * @tparam Type The pointer type (automatically deduced)
164 * @param R Pointer to check
165 * @return true if R points to an object of type ThisType or derived from it
166 *
167 * @note This method is used internally by:
168 * - corelib::isa<T>() for type checking
169 * - corelib::cast<T>() for checked casting
170 * - corelib::dyn_cast<T>() for safe casting
171 *
172 * Example usage through casting functions:
173 * @code
174 * Base *ptr = new MyType();
175 * if (isa<MyType>(ptr)) {
176 * MyType *typed = cast<MyType>(ptr); // Uses classof internally
177 * }
178 * @endcode
179 */
180 template <typename Type>
181 static bool classof(const Type *R) {
182 CRLB_ZONE_SCOPED;
183 return R->template IsA<ThisType>();
184 }
185};
