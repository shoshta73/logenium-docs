20/**
21 * @ingroup casting
22 * @brief Checked cast with assertion on type mismatch.
23 *
24 * cast() performs a type-safe downcast with runtime type checking and assertion.
25 * It first checks if the cast is valid using isa<To>(Val), then performs the cast.
26 * If the type check fails, it asserts in debug builds.
27 *
28 * Supported types:
29 * - References (const and non-const)
30 * - Pointers
31 * - std::unique_ptr (transfers ownership)
32 *
33 * Return type preserves const-correctness and reference/pointer semantics:
34 * - const From& → const To&
35 * - From& → To&
36 * - From* → To*
37 * - unique_ptr<From> → unique_ptr<To>
38 *
39 * @tparam To The target type
40 * @tparam From The source type (deduced)
41 * @param Val The value to cast
42 * @return The casted value (type depends on input)
43 *
44 * @note Asserts if Val is not an instance of To (debug::Assert)
45 * @note For pointers/unique_ptr, asserts Val is not null
46 * @note For unique_ptr, ownership is transferred (Val becomes empty)
47 * @note Uses decltype(auto) to preserve value categories
48 *
49 * Example:
50 * @code
51 * Base *base = getDerived();
52 *
53 * // Cast pointer (asserts if not Derived)
54 * Derived *derived = cast<Derived>(base);
55 *
56 * // Cast reference
57 * Base &ref = ...;
58 * Derived &dref = cast<Derived>(ref);
59 *
60 * // Cast unique_ptr (transfers ownership)
61 * std::unique_ptr<Base> uptr = ...;
62 * std::unique_ptr<Derived> dptr = cast<Derived>(std::move(uptr));
63 * // uptr is now empty
64 *
65 * // Preserves const
66 * const Base *cbase = ...;
67 * const Derived *cderived = cast<Derived>(cbase);
68 * @endcode
69 *
70 * @see isa() for type checking without casting
71 * @see dyn_cast() for casting that returns nullptr on failure
72 * @see cast_if_present() for null-safe casting
73 */
74template <typename To, typename From>
75inline decltype(auto) cast(const From &Val) {
76 CRLB_ZONE_SCOPED;
77 debug::Assert(isa<To>(Val), "cast<{}>(const {} &) arguments of incompatible type!", type_name<To>(),
78 type_name<From>());
79 return detail::CastInfo<To, const From>::DoCast(Val);
80}

107/**
108 * @ingroup casting
109 * @brief Null-safe checked cast with assertion on type mismatch.
110 *
111 * cast_if_present() combines presence checking with checked casting. It first
112 * checks if Val is present (not null/empty), returning a null value if absent.
113 * If present, it unwraps the value and performs cast<X>() with assertion.
114 *
115 * Useful for optional types and nullable pointers where you want null-safety
116 * but still want assertion on type mismatches for non-null values.
117 *
118 * Supported types:
119 * - std::optional (checks has_value())
120 * - Pointers (checks != nullptr)
121 * - std::unique_ptr (checks != nullptr, transfers ownership)
122 * - Regular types (always present)
123 *
124 * @tparam X The target type
125 * @tparam Y The source type (deduced)
126 * @param Val The value to check and cast
127 * @return Casted value if present and correct type, null/empty otherwise
128 *
129 * @note Returns failure value if Val is not present (no assertion)
130 * @note Asserts if Val is present but not of type X (debug::Assert)
131 * @note For unique_ptr, only transfers ownership if present and correct type
132 *
133 * Example:
134 * @code
135 * std::optional<Base> opt = ...;
136 *
137 * // Returns std::nullopt if opt is empty, asserts if wrong type
138 * auto result = cast_if_present<Derived>(opt);
139 *
140 * Base *ptr = ...; // May be nullptr
141 *
142 * // Returns nullptr if ptr is null, asserts if wrong type
143 * Derived *dptr = cast_if_present<Derived>(ptr);
144 *
145 * std::unique_ptr<Base> uptr = ...;
146 * auto duptr = cast_if_present<Derived>(std::move(uptr));
147 * // If uptr was null: duptr is null, uptr unchanged
148 * // If uptr was non-null and correct type: ownership transferred
149 * @endcode
150 *
151 * @see cast() for casting with assertion (no null check)
152 * @see isa_and_present() for checking type and presence
153 * @see cast_or_null() for alias to this function
154 */
155template <class X, class Y>
156inline auto cast_if_present(const Y &Val) {
157 CRLB_ZONE_SCOPED;
158 if (!detail::isPresent(Val)) return detail::CastInfo<X, const Y>::CastFailed();
159 debug::Assert(isa<X>(Val), "cast_if_present<{}>(const {} &) arguments of incompatible type!", type_name<X>(),
160 type_name<Y>());
161 return cast<X>(detail::unwrapValue(Val));
162}

192/**
193 * @ingroup casting
194 * @brief Alias for cast_if_present() - null-safe checked cast.
195 *
196 * cast_or_null() is an alias for cast_if_present(), providing a more intuitive
197 * name for pointer-based casting. It returns nullptr if the cast fails or if
198 * the input is null.
199 *
200 * @tparam X The target type
201 * @tparam Y The source type (deduced)
202 * @param Val The value to check and cast
203 * @return Casted value if present and correct type, null/empty otherwise
204 *
205 * @note Identical to cast_if_present()
206 * @note Asserts if Val is non-null but not of type X
207 *
208 * Example:
209 * @code
210 * Base *base = ...; // May be nullptr
211 *
212 * // Returns nullptr if base is null or wrong type (with assertion)
213 * Derived *derived = cast_or_null<Derived>(base);
214 *
215 * if (derived) {
216 * // Safe to use derived
217 * }
218 * @endcode
219 *
220 * @see cast_if_present() for the underlying implementation
221 */
222template <class X, class Y>
223auto cast_or_null(const Y &Val) {
224 return cast_if_present<X>(Val);
225}
