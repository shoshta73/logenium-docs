13/**
14 * @ingroup utility
15 * @brief RAII wrapper for executing a functor at scope exit.
16 *
17 * Defer is a simple RAII utility that executes a given functor when the Defer
18 * object is destroyed (at scope exit). This is useful for cleanup operations,
19 * ensuring resources are released even in the presence of early returns or
20 * exceptions.
21 *
22 * @tparam Functor The type of the callable object (lambda, function object, etc.)
23 *
24 * @note Multiple Defer objects in the same scope execute in LIFO order (reverse
25 * construction order)
26 * @note This class is move-only (not copyable)
27 *
28 * Example:
29 * @code
30 * void process_file(const char *filename) {
31 * FILE *file = fopen(filename, "r");
32 * if (!file) return;
33 *
34 * // Automatically close file at scope exit
35 * Defer cleanup([file]() { fclose(file); });
36 *
37 * // ... use file ...
38 * // file is automatically closed here, even if exceptions are thrown
39 * }
40 *
41 * // Multiple defers execute in reverse order:
42 * {
43 * Defer d1([]() { std::cout << "3\n"; });
44 * Defer d2([]() { std::cout << "2\n"; });
45 * Defer d3([]() { std::cout << "1\n"; });
46 * // Prints: 1, 2, 3
47 * }
48 * @endcode
49 */
50template <typename Functor>
51struct Defer {
52 /**
53 * @brief Construct a Defer object with a functor to execute at scope exit.
54 *
55 * @param functor The callable object to execute in the destructor (moved).
56 * Should be noexcept or handle exceptions internally to avoid
57 * std::terminate when called from the destructor.
58 */
59 constexpr Defer(Functor &&functor) : functor(std::move(functor)) {}
60
61 /**
62 * @brief Destructor that executes the stored functor.
63 */
64 ~Defer() {
65 CRLB_ZONE_SCOPED;
66 functor();
67 }
68
69 private:
70 /**
71 * @brief The callable object to execute at scope exit.
72 *
73 * @note Should be noexcept or handle exceptions internally to avoid
74 * std::terminate when called from the destructor.
75 */
76 Functor functor;
77};
