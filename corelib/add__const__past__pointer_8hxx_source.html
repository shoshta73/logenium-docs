11/**
12 * @ingroup type_traits
13 * @brief Type trait to add const qualification past pointer indirection.
14 *
15 * This trait adds const qualification to a type, but if the type is a pointer,
16 * it adds const to the pointee (the type being pointed to) rather than to the
17 * pointer itself.
18 *
19 * Behavior:
20 * - For pointer types: `T*` → `const T*` (const pointee)
21 * - For non-pointer types: `T` → `const T` (const type)
22 *
23 * This is different from `std::add_const` which would produce:
24 * - `T*` → `T* const` (const pointer, non-const pointee)
25 *
26 * @tparam T The type to add const to
27 *
28 * ## Member Types
29 *
30 * - `type`: The resulting type with const added past pointer indirection
31 *
32 * ## Helper Type Alias
33 *
34 * `add_const_past_pointer_t<T>` provides a convenient way to access the type:
35 * @code
36 * using T = add_const_past_pointer_t<int*>; // const int*
37 * @endcode
38 *
39 * @note This trait is primarily used internally by the casting module to preserve
40 * const-correctness when working with pointer types
41 *
42 * Example:
43 * @code
44 * // Pointer types - adds const to pointee
45 * static_assert(std::is_same_v<add_const_past_pointer_t<int*>, const int*>);
46 * static_assert(std::is_same_v<add_const_past_pointer_t<char*>, const char*>);
47 * static_assert(std::is_same_v<add_const_past_pointer_t<void*>, const void*>);
48 *
49 * // Already const pointee - no change
50 * static_assert(std::is_same_v<add_const_past_pointer_t<const int*>, const int*>);
51 *
52 * // Non-pointer types - adds const to type itself
53 * static_assert(std::is_same_v<add_const_past_pointer_t<int>, const int>);
54 * static_assert(std::is_same_v<add_const_past_pointer_t<double>, const double>);
55 *
56 * // Compare with std::add_const (different behavior for pointers)
57 * static_assert(std::is_same_v<std::add_const_t<int*>, int* const>); // const pointer
58 * static_assert(std::is_same_v<add_const_past_pointer_t<int*>, const int*>); // const pointee
59 * @endcode
60 */
61template <typename T>
62struct add_const_past_pointer {
63 using type = std::conditional_t<std::is_pointer_v<T>, const std::remove_pointer_t<T> *, const T>;
64};
65
66/**
67 * @ingroup type_traits
68 * @brief Helper type alias for add_const_past_pointer.
69 *
70 * Provides convenient access to add_const_past_pointer<T>::type following
71 * the C++14 standard library convention.
72 *
73 * @tparam T The type to add const to
74 *
75 * Example:
76 * @code
77 * template <typename T>
78 * void foo(add_const_past_pointer_t<T> param) {
79 * // param has const added past pointer indirection
80 * }
81 * @endcode
82 */
83template <typename T>
84using add_const_past_pointer_t = typename add_const_past_pointer<T>::type;
