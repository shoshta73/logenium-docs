21/**
22 * @ingroup casting_detail
23 * @brief Base template for recursive type simplification in the casting system.
24 *
25 * SimplifyType provides a mechanism to recursively unwrap type qualifiers
26 * (particularly const) to find the underlying "simple" type. This is a core
27 * component of the casting infrastructure, enabling const-correct casts through
28 * layers of type qualifiers.
29 *
30 * The base template represents the identity case - a type that is already
31 * "simple" (has no additional qualifiers to strip). The specialization for
32 * const types (SimplifyType<const From>) performs the actual recursive unwrapping.
33 *
34 * @tparam From The type to simplify
35 *
36 * ## Member Types
37 *
38 * - `SimpleType`: The simplified version of From (in base template, From itself)
39 *
40 * ## Static Methods
41 *
42 * - `GetSimplifiedValue(From &Val)`: Returns a reference to the simplified value
43 *
44 * @todo Rename this and/or replace it with other cast traits
45 *
46 * @note This is an internal detail type trait used by the casting infrastructure
47 * @note The const specialization performs recursive unwrapping through type layers
48 *
49 * Example:
50 * @code
51 * // Base case - type is already simple
52 * static_assert(std::is_same_v<SimplifyType<int>::SimpleType, int>);
53 * static_assert(std::is_same_v<SimplifyType<int*>::SimpleType, int*>);
54 *
55 * // Const unwrapping happens in the specialization
56 * // SimplifyType<const int>::SimpleType → const int (via const specialization)
57 * // SimplifyType<const int*>::SimpleType → const int* (pointer to const)
58 * @endcode
59 *
60 * @see SimplifyType<const From> for const unwrapping behavior
61 * @see IsSimpleType for checking if a type is already simplified
62 */
63template <typename From>
64struct SimplifyType {
65 using SimpleType = From;
66
67 static SimpleType &GetSimplifiedValue(From &Val) { return Val; }
68};

70/**
71 * @ingroup casting_detail
72 * @brief Specialization for const-qualified types that recursively unwraps const.
73 *
74 * This specialization handles const-qualified types by recursively applying
75 * SimplifyType to the non-const type, then re-applying const correctly based
76 * on whether the simplified type is a pointer or not.
77 *
78 * The unwrapping process:
79 * 1. Strip const from From to get the underlying type
80 * 2. Recursively simplify the underlying type
81 * 3. Re-apply const past pointer indirection (for pointers, constifies pointee)
82 * 4. Add lvalue reference if not a pointer
83 *
84 * @tparam From The const-qualified type to simplify
85 *
86 * ## Member Types
87 *
88 * - `NonConstSimpleType`: The simplified type without const
89 * - `SimpleType`: NonConstSimpleType with const re-applied past pointers
90 * - `RetType`: SimpleType with lvalue reference added (if not a pointer)
91 *
92 * ## Static Methods
93 *
94 * - `GetSimplifiedValue(const From &Val)`: Returns a reference to the simplified
95 * value, preserving const-correctness by recursively delegating to the
96 * non-const SimplifyType and casting away const safely
97 *
98 * @note Uses add_const_past_pointer_t to preserve const-correctness for pointers
99 * @note Uses add_lvalue_reference_if_not_pointer_t for non-pointer return types
100 *
101 * Example:
102 * @code
103 * // Const on non-pointer types
104 * static_assert(std::is_same_v<SimplifyType<const int>::SimpleType, const int>);
105 *
106 * // Const on pointer types - const applies to pointee
107 * static_assert(std::is_same_v<SimplifyType<const int*>::SimpleType, const int*>);
108 *
109 * // Multiple levels of const unwrapping
110 * // SimplifyType<const int*> → SimpleType = const int*
111 * @endcode
112 *
113 * @see add_const_past_pointer_t for const propagation behavior
114 * @see add_lvalue_reference_if_not_pointer_t for reference addition
115 */
116template <typename From>
117struct SimplifyType<const From> {
118 using NonConstSimpleType = typename SimplifyType<From>::SimpleType;
119 using SimpleType = add_const_past_pointer_t<NonConstSimpleType>;
120 using RetType = add_lvalue_reference_if_not_pointer_t<SimpleType>;
121
122 static RetType GetSimplifiedValue(const From &Val) {
123 CRLB_ZONE_SCOPED;
124 return SimplifyType<From>::GetSimplifiedValue(const_cast<From &>(Val));
125 }
126};

128/**
129 * @ingroup casting_detail
130 * @brief Type trait to check if a type is already in simplified form.
131 *
132 * IsSimpleType checks whether a type X is equal to its SimplifyType<X>::SimpleType.
133 * A type is considered "simple" when SimplifyType produces the same type,
134 * indicating no further simplification is possible.
135 *
136 * This is useful for optimizations and compile-time checks in the casting
137 * infrastructure to determine if type simplification is needed.
138 *
139 * @tparam X The type to check
140 *
141 * ## Member Constants
142 *
143 * - `value`: true if X is already simplified (X == SimplifyType<X>::SimpleType),
144 * false otherwise
145 *
146 * @note This is an internal trait used by the casting system
147 *
148 * Example:
149 * @code
150 * // Already simple types
151 * static_assert(IsSimpleType<int>::value);
152 * static_assert(IsSimpleType<int*>::value);
153 * static_assert(IsSimpleType<const int*>::value);
154 *
155 * // Types requiring simplification would be detected here
156 * // (behavior depends on SimplifyType specializations)
157 * @endcode
158 *
159 * @see SimplifyType for the simplification mechanism
160 */
161template <class X>
162struct IsSimpleType {
163 static const bool value = std::is_same_v<X, typename SimplifyType<X>::SimpleType>;
164};

170/**
171 * @ingroup casting_detail
172 * @brief Variable template to detect if a type can represent a null/empty state.
173 *
174 * IsNullable checks whether a type T can be constructed from or compared with
175 * nullptr, indicating it has nullable semantics. This includes:
176 * - All pointer types (T*)
177 * - Types constructible from std::nullptr_t (like std::unique_ptr, smart pointers)
178 *
179 * This trait is used by the casting system to determine if a type supports
180 * null-safe operations like dyn_cast_or_null() and cast_or_null().
181 *
182 * @tparam T The type to check for nullable semantics
183 *
184 * @note Returns true for raw pointers and smart pointers
185 * @note Returns false for value types and references
186 * @note Used internally by ValueIsPresent to specialize for nullable types
187 *
188 * Example:
189 * @code
190 * // Pointer types - nullable
191 * static_assert(IsNullable<int*>);
192 * static_assert(IsNullable<const char*>);
193 * static_assert(IsNullable<void*>);
194 *
195 * // Smart pointers - nullable (constructible from nullptr)
196 * static_assert(IsNullable<std::unique_ptr<int>>);
197 *
198 * // Value types - not nullable
199 * static_assert(!IsNullable<int>);
200 * static_assert(!IsNullable<double>);
201 * static_assert(!IsNullable<std::string>);
202 *
203 * // std::optional is not considered "nullable" in this sense
204 * // (it's not constructible from nullptr, uses std::nullopt instead)
205 * static_assert(!IsNullable<std::optional<int>>);
206 * @endcode
207 *
208 * @see ValueIsPresent for how nullable types are handled in presence checking
209 */
210template <typename T>
211constexpr bool IsNullable = std::is_pointer_v<T> || std::is_constructible_v<T, std::nullptr_t>;

217/**
218 * @ingroup casting_detail
219 * @brief Base template for detecting value presence and unwrapping values.
220 *
221 * ValueIsPresent provides a uniform interface for checking if a value is
222 * "present" (not null, not empty) and for unwrapping container types to
223 * access the underlying value. This is a core component enabling null-safe
224 * casting operations.
225 *
226 * The base template handles regular (non-nullable, non-optional) types,
227 * where values are always present. Specializations handle std::optional
228 * and nullable types (pointers, types constructible from nullptr).
229 *
230 * @tparam T The type to check for presence
231 * @tparam Enable SFINAE enabler for specializations
232 *
233 * ## Member Types
234 *
235 * - `UnwrappedType`: The underlying value type (T for base template)
236 *
237 * ## Static Methods
238 *
239 * - `IsPresent(const T &t)`: Always returns true for regular types
240 * - `UnwrapValue(T &t)`: Returns the value unchanged (no unwrapping needed)
241 *
242 * @note Specializations exist for std::optional and nullable types
243 * @note Used by cast_if_present(), dyn_cast_if_present(), etc.
244 *
245 * Example:
246 * @code
247 * // Regular types - always present
248 * int x = 42;
249 * static_assert(ValueIsPresent<int>::IsPresent(x));
250 * static_assert(ValueIsPresent<int>::UnwrapValue(x) == 42);
251 *
252 * std::string s = "hello";
253 * static_assert(ValueIsPresent<std::string>::IsPresent(s));
254 * @endcode
255 *
256 * @see ValueIsPresent<std::optional<T>> for optional value handling
257 * @see ValueIsPresent<T, enable_if nullable> for nullable type handling
258 */
259template <typename T, typename Enable = void>
260struct ValueIsPresent {
261 using UnwrappedType = T;
262 static inline bool IsPresent(const T &t) {
263 CRLB_ZONE_SCOPED;
264 return true;
265 }
266 static inline decltype(auto) UnwrapValue(T &t) {
267 CRLB_ZONE_SCOPED;
268 return t;
269 }
270};

272/**
273 * @ingroup casting_detail
274 * @brief Specialization for std::optional<T> that checks has_value() and unwraps.
275 *
276 * This specialization handles std::optional types by:
277 * - Checking presence via has_value()
278 * - Unwrapping via operator* to access the contained value
279 *
280 * @tparam T The type contained in the optional
281 *
282 * ## Member Types
283 *
284 * - `UnwrappedType`: The contained type T (not std::optional<T>)
285 *
286 * ## Static Methods
287 *
288 * - `IsPresent(const std::optional<T> &t)`: Returns t.has_value()
289 * - `UnwrapValue(std::optional<T> &t)`: Returns *t (the contained value)
290 *
291 * @note Enables cast_if_present() and dyn_cast_if_present() to work with optionals
292 * @note UnwrapValue assumes the optional contains a value (call IsPresent first)
293 *
294 * Example:
295 * @code
296 * std::optional<int> opt1 = 42;
297 * std::optional<int> opt2 = std::nullopt;
298 *
299 * static_assert(ValueIsPresent<std::optional<int>>::IsPresent(opt1)); // true
300 * static_assert(!ValueIsPresent<std::optional<int>>::IsPresent(opt2)); // false
301 *
302 * auto &value = ValueIsPresent<std::optional<int>>::UnwrapValue(opt1);
303 * // value is int& = 42
304 * @endcode
305 */
306template <typename T>
307struct ValueIsPresent<std::optional<T>> {
308 using UnwrappedType = T;
309 static inline bool IsPresent(const std::optional<T> &t) {
310 CRLB_ZONE_SCOPED;
311 return t.has_value();
312 }
313 static inline decltype(auto) UnwrapValue(std::optional<T> &t) {
314 CRLB_ZONE_SCOPED;
315 return *t;
316 }
317};

319/**
320 * @ingroup casting_detail
321 * @brief Specialization for nullable types (pointers and nullptr-constructible types).
322 *
323 * This specialization handles types that can be null-checked against nullptr:
324 * - Raw pointers (int*, char*, etc.)
325 * - Smart pointers (std::unique_ptr, types constructible from nullptr)
326 *
327 * Presence is checked by comparing against T(nullptr).
328 *
329 * @tparam T The nullable type (must satisfy IsNullable<T>)
330 *
331 * ## Member Types
332 *
333 * - `UnwrappedType`: The nullable type T itself (no unwrapping for pointers)
334 *
335 * ## Static Methods
336 *
337 * - `IsPresent(const T &t)`: Returns true if t != T(nullptr)
338 * - `UnwrapValue(T &t)`: Returns the value unchanged (pointers don't need unwrapping)
339 *
340 * @note Enabled via SFINAE when IsNullable<T> is true
341 * @note For pointers, UnwrappedType is still the pointer type (dereference happens elsewhere)
342 * @note Enables cast_or_null() and dyn_cast_or_null() to work with pointers
343 *
344 * Example:
345 * @code
346 * int x = 42;
347 * int *ptr1 = &x;
348 * int *ptr2 = nullptr;
349 *
350 * static_assert(ValueIsPresent<int*>::IsPresent(ptr1)); // true
351 * static_assert(!ValueIsPresent<int*>::IsPresent(ptr2)); // false
352 *
353 * auto unwrapped = ValueIsPresent<int*>::UnwrapValue(ptr1);
354 * // unwrapped is int* (same as ptr1, no dereferencing)
355 *
356 * std::unique_ptr<int> uptr = std::make_unique<int>(42);
357 * static_assert(ValueIsPresent<std::unique_ptr<int>>::IsPresent(uptr));
358 * @endcode
359 *
360 * @see IsNullable for the nullable type detection trait
361 */
362template <typename T>
363struct ValueIsPresent<T, std::enable_if_t<detail::IsNullable<T>>> {
364 using UnwrappedType = T;
365 static inline bool IsPresent(const T &t) {
366 CRLB_ZONE_SCOPED;
367 return t != T(nullptr);
368 }
369 static inline decltype(auto) UnwrapValue(T &t) {
370 CRLB_ZONE_SCOPED;
371 return t;
372 }
373};

375/**
376 * @ingroup casting_detail
377 * @brief Helper function to check if a value is present (not null/empty).
378 *
379 * This function provides a convenient interface to ValueIsPresent::IsPresent()
380 * that also handles type simplification. It first simplifies the type T
381 * (unwrapping const qualifiers), then checks presence on the simplified value.
382 *
383 * Works with:
384 * - Regular types (always present)
385 * - std::optional (checks has_value())
386 * - Pointers and smart pointers (checks != nullptr)
387 *
388 * @tparam T The type to check (const qualifiers are automatically handled)
389 * @param t The value to check for presence
390 * @return true if the value is present (not null/empty), false otherwise
391 *
392 * @note Automatically handles const-qualified types via SimplifyType
393 * @note Used internally by cast_if_present() and dyn_cast_if_present()
394 *
395 * Example:
396 * @code
397 * int x = 42;
398 * static_assert(isPresent(x)); // true - regular values always present
399 *
400 * int *ptr = &x;
401 * static_assert(isPresent(ptr)); // true - non-null pointer
402 *
403 * int *null_ptr = nullptr;
404 * static_assert(!isPresent(null_ptr)); // false - null pointer
405 *
406 * std::optional<int> opt = 42;
407 * static_assert(isPresent(opt)); // true - optional has value
408 *
409 * std::optional<int> empty;
410 * static_assert(!isPresent(empty)); // false - optional is empty
411 * @endcode
412 *
413 * @see ValueIsPresent for the underlying presence checking mechanism
414 * @see unwrapValue() for extracting the value after checking presence
415 */
416template <typename T>
417inline bool isPresent(const T &t) {
418 CRLB_ZONE_SCOPED;
419 return ValueIsPresent<typename SimplifyType<T>::SimpleType>::IsPresent(
420 SimplifyType<T>::GetSimplifiedValue(const_cast<T &>(t)));
421}

423/**
424 * @ingroup casting_detail
425 * @brief Helper function to unwrap a value from its container.
426 *
427 * This function provides a convenient interface to ValueIsPresent::UnwrapValue().
428 * For most types, this is an identity operation (returns the value unchanged).
429 * For std::optional, it dereferences to get the contained value.
430 *
431 * @tparam T The type to unwrap
432 * @param t The value to unwrap
433 * @return The unwrapped value (decltype(auto) preserves reference type)
434 *
435 * @note For std::optional, assumes the value is present (call isPresent() first)
436 * @note For pointers, returns the pointer unchanged (no dereferencing)
437 * @note Uses decltype(auto) to preserve value category and reference type
438 *
439 * Example:
440 * @code
441 * std::optional<int> opt = 42;
442 * auto &value = unwrapValue(opt);
443 * // value is int& = 42
444 *
445 * int x = 10;
446 * auto &same = unwrapValue(x);
447 * // same is int& = 10 (identity operation)
448 *
449 * int *ptr = &x;
450 * auto p = unwrapValue(ptr);
451 * // p is int* (pointer unchanged, not dereferenced)
452 * @endcode
453 *
454 * @see ValueIsPresent for the underlying unwrapping mechanism
455 * @see isPresent() for checking if a value should be unwrapped
456 */
457template <typename T>
458inline decltype(auto) unwrapValue(T &t) {
459 CRLB_ZONE_SCOPED;
460 return ValueIsPresent<T>::UnwrapValue(t);
461}

467/**
468 * @ingroup casting_detail
469 * @brief Helper type alias for CRTP pattern with optional derived class override.
470 *
471 * SelfType implements a common CRTP (Curiously Recurring Template Pattern)
472 * idiom where a base class template takes an optional derived type parameter.
473 * If the derived type is void, it defaults to a specified default type.
474 *
475 * This is used extensively in the casting infrastructure to allow customization
476 * points while providing sensible defaults. Template classes can use SelfType
477 * to refer to either a user-provided derived class or themselves.
478 *
479 * @tparam OptionalDerived The optional derived class (or void for default)
480 * @tparam Default The default type to use when OptionalDerived is void
481 *
482 * Pattern:
483 * @code
484 * // Base class using SelfType for CRTP
485 * template <typename To, typename From, typename Derived = void>
486 * struct CastOperation {
487 * using Self = SelfType<Derived, CastOperation<To, From>>;
488 *
489 * // Methods can call static_cast<Self*>(this) to access derived overrides
490 * };
491 *
492 * // User can provide custom derived class
493 * struct MyCustomCast : CastOperation<int, double, MyCustomCast> {
494 * // Override behavior here
495 * };
496 *
497 * // Or use the default (Derived = void)
498 * using DefaultCast = CastOperation<int, double>;
499 * @endcode
500 *
501 * @note Used by casting infrastructure classes like DefaultDoCastIfPossible
502 * @note Enables customization without requiring explicit derived classes
503 *
504 * Example:
505 * @code
506 * // When OptionalDerived is a type
507 * static_assert(std::is_same_v<SelfType<MyClass, DefaultClass>, MyClass>);
508 *
509 * // When OptionalDerived is void
510 * static_assert(std::is_same_v<SelfType<void, DefaultClass>, DefaultClass>);
511 * @endcode
512 *
513 * @see DefaultDoCastIfPossible for usage in casting infrastructure
514 */
515template <typename OptionalDerived, typename Default>
516using SelfType = std::conditional_t<std::is_same_v<OptionalDerived, void>, Default, OptionalDerived>;
