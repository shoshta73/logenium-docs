14/**
15 * @ingroup casting_detail
16 * @brief Predicate functor for variadic type checking.
17 *
18 * IsaCheckPredicate wraps the isa<Types...>() function in a functor suitable
19 * for use with standard algorithms. It checks if a value is an instance of
20 * any of the specified types.
21 *
22 * @tparam Types The target types to check for (variadic)
23 *
24 * @note Used internally by public IsaPred variable template
25 *
26 * Example:
27 * @code
28 * std::vector<Base*> ptrs = ...;
29 * IsaCheckPredicate<Derived> pred;
30 *
31 * auto it = std::find_if(ptrs.begin(), ptrs.end(), pred);
32 * // Finds first element that is a Derived*
33 * @endcode
34 *
35 * @see isa() for the underlying type-checking function
36 * @see IsaPred for the public variable template interface
37 */
38template <typename... Types>
39struct IsaCheckPredicate {
40 template <typename T>
41 bool operator()(const T &Val) const {
42 CRLB_ZONE_SCOPED;
43 return isa<Types...>(Val);
44 }
45};
