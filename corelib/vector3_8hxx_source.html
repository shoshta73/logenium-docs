15/**
16 * @ingroup math_vectors
17 * @brief Template-based 3D vector class.
18 *
19 * Vec3 represents a three-dimensional vector with components accessible
20 * through multiple naming schemes (x/y/z, r/g/b, s/t/p) for different use cases.
21 *
22 * @tparam T The component type (float, double, int, etc.)
23 *
24 * @note Most operations are constexpr and can be evaluated at compile time
25 * @note Length() and Normalized() are non-constexpr due to std::sqrt
26 * @note Provides Cross() product method, unique to 3D vectors
27 *
28 * Example:
29 * @code
30 * Vec3<float> a{1.0f, 2.0f, 3.0f};
31 * Vec3<float> b{4.0f, 5.0f, 6.0f};
32 * auto sum = a + b; // (5, 7, 9)
33 * float dot = Vec3<float>::Dot(a, b); // 32
34 * Vec3<float> cross = Vec3<float>::Cross(a, b); // (-3, 6, -3)
35 * Vec3<float> unit = a.Normalized(); // (0.267, 0.535, 0.802)
36 * @endcode
37 */
38template <typename T>
39struct vec3 {
40 /**
41 * @brief Union providing multiple naming schemes for the components.
42 *
43 * - x, y, z: Spatial coordinates
44 * - r, g, b: Color (red, green, blue) components
45 * - s, t, p: Texture coordinates
46 */
47 union {
48 struct {
49 T x;
50 T y;
51 T z;
52 };
53 struct {
54 T r;
55 T g;
56 T b;
57 };
58 struct {
59 T s;
60 T t;
61 T p;
62 };
63 };
64
65 /**
66 * @brief Default constructor, initializes all components to zero.
67 */
68 constexpr vec3() : x(T{0}), y(T{0}), z(T{0}) { CRLB_ZONE_SCOPED; }
69
70 /**
71 * @brief Construct from a uniform value.
72 * @param val The value for all components
73 */
74 constexpr vec3(T val) : x(val), y(val), z(val) { CRLB_ZONE_SCOPED; }
75
76 /**
77 * @brief Construct from individual components.
78 * @param x The x component
79 * @param y The y component
80 * @param z The z component
81 */
82 constexpr vec3(T x, T y, T z) : x(x), y(y), z(z) { CRLB_ZONE_SCOPED; }
83
84 /**
85 * @brief Create a zero vector (0, 0, 0).
86 * @return Vec3 with all components set to zero
87 */
88 static constexpr vec3 Zero() {
89 CRLB_ZONE_SCOPED;
90 return {T{0}, T{0}, T{0}};
91 }
92
93 /**
94 * @brief Create a unit vector (1, 1, 1).
95 * @return Vec3 with all components set to one
96 */
97 static constexpr vec3 One() { return {T{1}, T{1}, T{1}}; }
98
99 constexpr vec3 operator+(const vec3 &other) const {
100 CRLB_ZONE_SCOPED;
101 return {x + other.x, y + other.y, z + other.z};
102 }
103 constexpr vec3 operator-(const vec3 &other) const {
104 CRLB_ZONE_SCOPED;
105 return {x - other.x, y - other.y, z - other.z};
106 }
107 constexpr vec3 operator*(const vec3 &other) const {
108 CRLB_ZONE_SCOPED;
109 return {x * other.x, y * other.y, z * other.z};
110 }
111 constexpr vec3 operator/(const vec3 &other) const {
112 CRLB_ZONE_SCOPED;
113 debug::Assert(other.x != T{0} && other.y != T{0} && other.z != T{0}, "Division by zero");
114 return {x / other.x, y / other.y, z / other.z};
115 }
116
117 constexpr vec3 &operator+=(const vec3 &other) {
118 CRLB_ZONE_SCOPED;
119 x += other.x;
120 y += other.y;
121 z += other.z;
122 return *this;
123 }
124 constexpr vec3 &operator-=(const vec3 &other) {
125 CRLB_ZONE_SCOPED;
126 x -= other.x;
127 y -= other.y;
128 z -= other.z;
129 return *this;
130 }
131
132 constexpr vec3 operator*(T scalar) const {
133 CRLB_ZONE_SCOPED;
134 return {x * scalar, y * scalar, z * scalar};
135 }
136 constexpr vec3 operator/(T scalar) const {
137 CRLB_ZONE_SCOPED;
138 debug::Assert(scalar != T{0}, "Division by zero");
139 return {x / scalar, y / scalar, z / scalar};
140 }
141
142 constexpr bool operator==(const vec3 &other) const {
143 CRLB_ZONE_SCOPED;
144 return x == other.x && y == other.y && z == other.z;
145 }
146 constexpr bool operator!=(const vec3 &other) const {
147 CRLB_ZONE_SCOPED;
148 return x != other.x || y != other.y || z != other.z;
149 }
150
151 /**
152 * @brief Calculate the squared length of the vector.
153 * @return The squared length (x * x + y * y + z * z)
154 * @note Faster than Length() as it avoids the square root operation
155 */
156 constexpr T LengthSquared() const {
157 CRLB_ZONE_SCOPED;
158 return x * x + y * y + z * z;
159 }
160
161 /**
162 * @brief Calculate the length (magnitude) of the vector.
163 * @return The Euclidean length sqrt(x * x + y * y + z * z)
164 * @note Non-constexpr due to std::sqrt
165 */
166 T Length() const {
167 CRLB_ZONE_SCOPED;
168 return std::sqrt(LengthSquared());
169 }
170
171 /**
172 * @brief Normalize the vector in-place to unit length.
173 * @return Reference to this vector after normalization
174 * @note Asserts if the vector has zero length (use debug::Assert)
175 */
176 vec3 &Normalize() {
177 CRLB_ZONE_SCOPED;
178 T len = Length();
179 debug::Assert(len != T{0}, "Cannot normalize a vector with zero length");
180 x /= len;
181 y /= len;
182 z /= len;
183 return *this;
184 }
185
186 /**
187 * @brief Get a normalized copy of this vector.
188 * @return A new Vec3 normalized to unit length
189 * @note Asserts if the vector has zero length (use debug::Assert)
190 * @note This method is constexpr
191 */
192 constexpr vec3 Normalized() const {
193 CRLB_ZONE_SCOPED;
194 T len = Length();
195 debug::Assert(len != T{0}, "Cannot normalize a vector with zero length");
196 return {x / len, y / len, z / len};
197 }
198
199 /**
200 * @brief Calculate the dot product of two vectors.
201 * @param a The first vector
202 * @param b The second vector
203 * @return The dot product (a.x * b.x + a.y * b.y + a.z * b.z)
204 */
205 static constexpr T Dot(const vec3 &a, const vec3 &b) {
206 CRLB_ZONE_SCOPED;
207 return a.x * b.x + a.y * b.y + a.z * b.z;
208 }
209
210 /**
211 * @brief Calculate the cross product of two vectors.
212 *
213 * The cross product produces a vector perpendicular to both input vectors,
214 * with magnitude equal to the area of the parallelogram formed by them.
215 *
216 * @param a The first vector
217 * @param b The second vector
218 * @return The cross product vector perpendicular to both a and b
219 * @note Order matters: Cross(a, b) = -Cross(b, a)
220 * @note The result follows the right-hand rule
221 */
222 static constexpr vec3 Cross(const vec3 &a, const vec3 &b) {
223 CRLB_ZONE_SCOPED;
224 return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
225 }
226};
