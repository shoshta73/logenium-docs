13/**
14 * @ingroup casting
15 * @brief Checks if a value is an instance of any of the specified types.
16 *
17 * isa() performs RTTI-free type checking using the classof() pattern. It checks
18 * whether Val is an instance of any of the types specified in the template
19 * parameter pack. Returns true if Val matches at least one type.
20 *
21 * This is the primary type-checking function in the casting system, supporting:
22 * - Variadic type checking (check multiple types with OR semantics)
23 * - References, pointers, const-qualified types
24 * - std::unique_ptr (asserts non-null)
25 * - LLVM-style classof() pattern (no C++ RTTI required)
26 *
27 * @tparam To The target type(s) to check for (variadic)
28 * @tparam From The source type (deduced from Val)
29 * @param Val The value to type-check
30 * @return true if Val is an instance of any type in To..., false otherwise
31 *
32 * @note Requires To types to have static classof() methods
33 * @note For pointers/unique_ptr, asserts Val is not null
34 * @note Uses fold expression with || for variadic checking
35 *
36 * Example:
37 * @code
38 * Base *base = ...;
39 *
40 * // Single type check
41 * if (isa<Derived>(base)) {
42 * // base points to a Derived instance
43 * }
44 *
45 * // Multiple type check (OR semantics)
46 * if (isa<Derived1, Derived2, Derived3>(base)) {
47 * // base is one of these types
48 * }
49 *
50 * // Works with references
51 * Base &ref = ...;
52 * if (isa<Derived>(ref)) { }
53 *
54 * // Works with const
55 * const Base *const_base = ...;
56 * if (isa<Derived>(const_base)) { }
57 * @endcode
58 *
59 * @see isa_and_present() for null-safe type checking
60 * @see cast() for type casting with assertion
61 * @see dyn_cast() for type casting with null on failure
62 */
63template <typename... To, typename From>
64inline bool isa(const From &Val) {
65 CRLB_ZONE_SCOPED;
66 return (detail::CastInfo<To, const From>::IsPossible(Val) || ...);
67}

69/**
70 * @ingroup casting
71 * @brief Null-safe type checking with presence validation.
72 *
73 * isa_and_present() combines presence checking with type checking. It first
74 * checks if Val is "present" (not null for pointers, has_value() for optional),
75 * then performs isa<X...>() if the value is present.
76 *
77 * This is useful for nullable types where you want to check both presence
78 * and type in a single operation, avoiding assertions from isa() on null values.
79 *
80 * @tparam X The target type(s) to check for (variadic)
81 * @tparam Y The source type (deduced from Val)
82 * @param Val The value to check
83 * @return true if Val is present AND is an instance of any type in X...,
84 * false if Val is null/empty or not the right type
85 *
86 * @note Short-circuits: returns false immediately if Val is not present
87 * @note Safe to call on null pointers or empty optionals (won't assert)
88 * @note For nullable types: checks != nullptr for pointers, has_value() for optional
89 *
90 * Example:
91 * @code
92 * Base *base = ...; // May be nullptr
93 *
94 * // Safe even if base is nullptr
95 * if (isa_and_present<Derived>(base)) {
96 * // base is non-null AND points to Derived
97 * }
98 *
99 * std::optional<Base> opt = ...;
100 * if (isa_and_present<Derived>(opt)) {
101 * // opt has value AND is Derived
102 * }
103 *
104 * // Variadic works too
105 * if (isa_and_present<Derived1, Derived2>(base)) {
106 * // base is non-null AND is one of these types
107 * }
108 * @endcode
109 *
110 * @see isa() for type checking (asserts on null)
111 * @see cast_if_present() for casting with presence check
112 * @see dyn_cast_if_present() for dynamic casting with presence check
113 */
114template <typename... X, class Y>
115inline bool isa_and_present(const Y &Val) {
116 CRLB_ZONE_SCOPED;
117 if (!detail::isPresent(Val)) return false;
118 return isa<X...>(Val);
119}
