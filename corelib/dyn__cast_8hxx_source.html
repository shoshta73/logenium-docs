19/**
20 * @ingroup casting
21 * @brief Dynamic cast that returns nullptr on type mismatch.
22 *
23 * dyn_cast() performs a type-safe downcast with runtime type checking.
24 * Unlike cast(), it returns a null value (nullptr for pointers, empty for
25 * unique_ptr) if the cast fails, rather than asserting. This provides a
26 * safe way to attempt casts where the type is uncertain.
27 *
28 * Supported types:
29 * - References (const and non-const) - returns reference or null reference (implementation defined)
30 * - Pointers - returns pointer or nullptr
31 * - std::unique_ptr - returns unique_ptr or empty unique_ptr
32 *
33 * Return type preserves const-correctness:
34 * - const From& → const To& or null
35 * - From& → To& or null
36 * - From* → To* or nullptr
37 * - unique_ptr<From>& → unique_ptr<To> or empty
38 *
39 * @tparam To The target type
40 * @tparam From The source type (deduced)
41 * @param Val The value to cast
42 * @return Casted value if type matches, nullptr/null value otherwise
43 *
44 * @note Asserts if Val is null (use dyn_cast_or_null for null-safe version)
45 * @note For pointers: returns nullptr on failure (safe to check)
46 * @note For unique_ptr: transfers ownership only if cast succeeds
47 * @note Uses decltype(auto) to preserve value categories
48 *
49 * Example:
50 * @code
51 * Base *base = getBase();
52 *
53 * // Safe cast - returns nullptr if not Derived
54 * if (Derived *derived = dyn_cast<Derived>(base)) {
55 * // base was a Derived, use it
56 * } else {
57 * // base was some other type
58 * }
59 *
60 * // Works with unique_ptr
61 * std::unique_ptr<Base> uptr = ...;
62 * if (auto dptr = dyn_cast<Derived>(uptr)) {
63 * // Cast succeeded, ownership transferred
64 * } else {
65 * // Cast failed, uptr still owns the object
66 * }
67 * @endcode
68 *
69 * @see cast() for checked cast with assertion
70 * @see dyn_cast_or_null() for null-safe version
71 * @see dyn_cast_if_present() for presence-checking version
72 */
73template <typename To, typename From>
74inline decltype(auto) dyn_cast(const From &Val) {
75 CRLB_ZONE_SCOPED;
76 debug::Assert(detail::isPresent(Val), "dyn_cast<{}>(const {} &) called on a non-existent value", type_name<To>(),
77 type_name<From>());
78 return detail::CastInfo<To, const From>::DoCastIfPossible(Val);
79}

108/**
109 * @ingroup casting
110 * @brief Null-safe dynamic cast with presence checking.
111 *
112 * dyn_cast_if_present() combines presence checking with dynamic casting.
113 * It first checks if Val is present (not null/empty), then performs dyn_cast()
114 * on the unwrapped value. Returns null/empty if Val is not present or if the
115 * cast fails.
116 *
117 * This is the safest casting operation - it never asserts and handles both
118 * null values and type mismatches gracefully.
119 *
120 * @tparam X The target type
121 * @tparam Y The source type (deduced)
122 * @param Val The value to check and cast
123 * @return Casted value if present and correct type, null/empty otherwise
124 *
125 * @note No assertions - safe to call on null/empty values
126 * @note Returns null if Val is not present (no type check performed)
127 * @note Returns null if Val is present but wrong type
128 *
129 * Example:
130 * @code
131 * Base *base = ...; // May be nullptr
132 *
133 * // Safe even if base is nullptr
134 * if (Derived *derived = dyn_cast_if_present<Derived>(base)) {
135 * // base was non-null AND was a Derived
136 * }
137 *
138 * std::optional<Base> opt = ...;
139 * if (auto derived = dyn_cast_if_present<Derived>(opt)) {
140 * // opt had value AND was Derived
141 * }
142 * @endcode
143 *
144 * @see dyn_cast() for dynamic cast (asserts on null)
145 * @see dyn_cast_or_null() for alias to this function
146 */
147template <class X, class Y>
148auto dyn_cast_if_present(const Y &Val) {
149 CRLB_ZONE_SCOPED;
150 if (!detail::isPresent(Val)) return detail::CastInfo<X, const Y>::CastFailed();
151 return detail::CastInfo<X, const Y>::DoCastIfPossible(detail::unwrapValue(Val));
152}
153
154/// @copydoc dyn_cast_if_present(const Y &)
155template <class X, class Y>
156auto dyn_cast_if_present(Y &Val) {
157 CRLB_ZONE_SCOPED;
158 if (!detail::isPresent(Val)) return detail::CastInfo<X, Y>::CastFailed();
159 return detail::CastInfo<X, Y>::DoCastIfPossible(detail::unwrapValue(Val));
160}

162/// @copydoc dyn_cast_if_present(const Y &)
163template <class X, class Y>
164auto dyn_cast_if_present(Y *Val) {
165 CRLB_ZONE_SCOPED;
166 if (!detail::isPresent(Val)) return detail::CastInfo<X, Y *>::CastFailed();
167 return detail::CastInfo<X, Y *>::DoCastIfPossible(detail::unwrapValue(Val));
168}

170/**
171 * @ingroup casting
172 * @brief Alias for dyn_cast_if_present() - null-safe dynamic cast.
173 *
174 * dyn_cast_or_null() is an alias for dyn_cast_if_present(), providing a more
175 * intuitive name for pointer-based casting. It returns nullptr if the cast
176 * fails or if the input is null.
177 *
178 * @tparam X The target type
179 * @tparam Y The source type (deduced)
180 * @param Val The value to check and cast
181 * @return Casted value if present and correct type, nullptr otherwise
182 *
183 * @note Identical to dyn_cast_if_present()
184 * @note No assertions - safe for null inputs and type mismatches
185 *
186 * Example:
187 * @code
188 * Base *base = ...; // May be nullptr
189 *
190 * // Returns nullptr if base is null OR wrong type
191 * Derived *derived = dyn_cast_or_null<Derived>(base);
192 * if (derived) {
193 * // Safe to use
194 * }
195 * @endcode
196 *
197 * @see dyn_cast_if_present() for the underlying implementation
198 */
199template <class X, class Y>
200auto dyn_cast_or_null(const Y &Val) {
201 return dyn_cast_if_present<X>(Val);
202}

216/**
217 * @ingroup casting
218 * @brief Specialized dynamic cast for std::unique_ptr with ownership transfer.
219 *
220 * unique_dyn_cast() performs a type-safe cast of unique_ptr with ownership
221 * semantics. If the cast succeeds, ownership is transferred to the returned
222 * unique_ptr. If the cast fails, the original unique_ptr retains ownership
223 * and an empty unique_ptr is returned.
224 *
225 * This function asserts that Val is non-null (use unique_dyn_cast_or_null
226 * for null-safe version).
227 *
228 * @tparam X The target type
229 * @tparam Y The source type (deduced)
230 * @param Val The unique_ptr to cast (passed by reference or rvalue)
231 * @return unique_ptr<X> if cast succeeds, empty unique_ptr otherwise
232 *
233 * @note Asserts if Val is null
234 * @note Ownership transferred only if cast succeeds
235 * @note Val becomes empty if cast succeeds, retains ownership if cast fails
236 *
237 * Example:
238 * @code
239 * std::unique_ptr<Base> base = getDerived();
240 *
241 * // Cast succeeds: ownership transferred
242 * if (auto derived = unique_dyn_cast<Derived>(base)) {
243 * // base is now empty, derived owns the object
244 * } else {
245 * // Cast failed, base still owns the object
246 * }
247 * @endcode
248 *
249 * @see dyn_cast() for general dynamic casting
250 * @see unique_dyn_cast_or_null() for null-safe version
251 */
252template <class X, class Y>
253inline typename detail::CastInfo<X, std::unique_ptr<Y>>::CastResultType unique_dyn_cast(std::unique_ptr<Y> &Val) {
254 CRLB_ZONE_SCOPED;
255 if (!isa<X>(Val)) return nullptr;
256 return cast<X>(std::move(Val));
257}
258
259/// @copydoc unique_dyn_cast(std::unique_ptr<Y> &)
260template <class X, class Y>
261inline auto unique_dyn_cast(std::unique_ptr<Y> &&Val) {
262 return unique_dyn_cast<X, Y>(Val);
263}

265/**
266 * @ingroup casting
267 * @brief Null-safe unique_ptr dynamic cast with ownership transfer.
268 *
269 * unique_dyn_cast_or_null() is the null-safe version of unique_dyn_cast().
270 * It checks if Val is null before attempting the cast. If null, returns nullptr
271 * immediately without assertion. Otherwise performs the dynamic cast.
272 *
273 * @tparam X The target type
274 * @tparam Y The source type (deduced)
275 * @param Val The unique_ptr to cast (may be null)
276 * @return unique_ptr<X> if non-null and cast succeeds, nullptr otherwise
277 *
278 * @note No assertion - safe to call on null unique_ptr
279 * @note Ownership transferred only if non-null and cast succeeds
280 *
281 * Example:
282 * @code
283 * std::unique_ptr<Base> base = ...; // May be nullptr
284 *
285 * // Safe even if base is nullptr
286 * if (auto derived = unique_dyn_cast_or_null<Derived>(base)) {
287 * // base was non-null AND was Derived, ownership transferred
288 * } else {
289 * // base was null OR wrong type
290 * if (base) {
291 * // base still owns the object (wrong type)
292 * }
293 * }
294 * @endcode
295 *
296 * @see unique_dyn_cast() for version that asserts on null
297 */
298template <class X, class Y>
299inline typename detail::CastInfo<X, std::unique_ptr<Y>>::CastResultType unique_dyn_cast_or_null(
300 std::unique_ptr<Y> &Val) {
301 CRLB_ZONE_SCOPED;
302 if (!Val) return nullptr;
303 return unique_dyn_cast<X, Y>(Val);
304}

308inline auto unique_dyn_cast_or_null(std::unique_ptr<Y> &&Val) {
309 return unique_dyn_cast_or_null<X, Y>(Val);
