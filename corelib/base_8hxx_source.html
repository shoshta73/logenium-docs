9namespace corelib::rtti {

11/**
12 * @brief Abstract base class for the extensible RTTI system.
13 * @ingroup extensible-rtti
14 *
15 * This class provides the foundation for a runtime type identification system
16 * that works without C++ RTTI. It uses unique type identifiers (addresses of
17 * static variables) to enable type checking and dynamic casting.
18 *
19 * The system is designed to work seamlessly with corelib's LLVM-style casting
20 * infrastructure (isa, cast, dyn_cast) through the classof pattern implemented
21 * in derived classes using the Extends template.
22 *
23 * @note Design Patterns Used:
24 * - Type ID Pattern: Each type has a unique static ID whose address serves
25 * as a compile-time constant type identifier
26 * - Anchor Idiom: Pure virtual anchor() method ensures vtable is emitted in
27 * the implementation file, preventing vtable/RTTI duplication across
28 * translation units
29 *
30 * @note Usage:
31 * Do not inherit from Base directly. Instead, use the Extends template which
32 * provides the necessary boilerplate.
33 *
34 * Example:
35 * @code
36 * class MyType : public Extends<MyType, Base> {
37 * public:
38 * MyType() {}
39 * private:
40 * static char ID;
41 * void anchor() override {}
42 * friend class Extends<MyType, Base>;
43 * };
44 *
45 * // In .cxx file:
46 * char MyType::ID = 0;
47 * @endcode
48 *
49 * @see Extends
50 */
51class Base {
52 public:
53 virtual ~Base() = default;
54
55 /**
56 * @brief Returns the static type identifier for this class.
57 *
58 * Each class in the hierarchy has a unique static ID variable. The address
59 * of this variable serves as a compile-time constant that uniquely identifies
60 * the type. This allows type checking without C++ RTTI.
61 *
62 * @return Pointer to the static ID variable for Base
63 */
64 static const void *TypeID() {
65 CRLB_ZONE_SCOPED;
66 return &ID;
67 }
68
69 /**
70 * @brief Returns the dynamic type identifier for the actual object type.
71 *
72 * This method must be overridden by all derived classes to return their
73 * specific TypeID(). The Extends template automatically provides this
74 * implementation.
75 *
76 * @return Pointer to the static ID variable of the most-derived type
77 */
78 [[nodiscard]] virtual const void *DynamicTypeID() const = 0;
79
80 /**
81 * @brief Checks if this object is an instance of or derived from the given type.
82 *
83 * This method performs runtime type checking by comparing type IDs. It
84 * checks if the object's dynamic type matches the queried type or any of
85 * its base types.
86 *
87 * The default implementation only checks against Base's TypeID. Derived
88 * classes using Extends will override this to check the entire inheritance
89 * chain.
90 *
91 * @param TID Type identifier to check against (obtained from TypeID())
92 * @return true if this object is of type TID or derives from it, false otherwise
93 *
94 * Example:
95 * @code
96 * Base *obj = new MyDerivedType();
97 * if (obj->IsA(MyDerivedType::TypeID())) {
98 * // obj is a MyDerivedType
99 * }
100 * @endcode
101 */
102 virtual bool IsA(const void *const TID) const {
103 CRLB_ZONE_SCOPED;
104 return TID == TypeID();
105 };
106
107 private:
108 /**
109 * @brief Anchor method to ensure vtable is emitted in implementation file.
110 *
111 * This pure virtual method forces derived classes to provide an implementation,
112 * which ensures the vtable is emitted in a single translation unit rather than
113 * being duplicated across multiple object files.
114 *
115 * @note Derived classes using Extends should provide an empty implementation:
116 * @code
117 * void anchor() override {}
118 * @endcode
119 */
120 virtual void anchor() = 0;
121
122 /**
123 * @brief Static type identifier for Base.
124 *
125 * The address of this variable serves as the unique identifier for the Base
126 * type. The actual value is irrelevant; only the address matters.
127 */
128 static char ID;
129};
130
131} // namespace corelib::rtti
