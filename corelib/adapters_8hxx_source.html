28/**
29 * @ingroup casting_detail
30 * @brief Casting adapter for std::unique_ptr with ownership transfer.
31 *
32 * UniquePtrCast enables casting of std::unique_ptr instances while preserving
33 * unique ownership semantics. When a cast succeeds, ownership is transferred
34 * from the source unique_ptr to the result. When a cast fails (in DoCastIfPossible),
35 * the source unique_ptr retains ownership.
36 *
37 * The implementation:
38 * 1. Checks if the cast is possible via CastIsPossible (inherited)
39 * 2. If possible, releases the source pointer and wraps it in a new unique_ptr
40 * 3. If not possible, returns nullptr unique_ptr (source unchanged)
41 *
42 * @tparam To The target type (for unique_ptr<To>)
43 * @tparam From The source type (from unique_ptr<From>)
44 * @tparam Derived Optional derived class for CRTP customization
45 *
46 * ## Member Types
47 *
48 * - `Self`: CRTP self type (Derived or UniquePtrCast<To, From>)
49 * - `CastResultType`: The result type (std::unique_ptr<To>)
50 *
51 * ## Static Methods
52 *
53 * - `DoCast(std::unique_ptr<From> &&f)`: Transfers ownership, returns unique_ptr<To>
54 * - `CastFailed()`: Returns empty unique_ptr<To>
55 * - `DoCastIfPossible(std::unique_ptr<From> &f)`: Checks possibility first,
56 * transfers ownership only if cast succeeds
57 *
58 * @note Inherits IsPossible from CastIsPossible<To, From*>
59 * @note DoCast unconditionally transfers ownership (caller must check first)
60 * @note DoCastIfPossible only transfers ownership if cast succeeds
61 * @note Uses CRTP pattern via SelfType for customization
62 *
63 * Example:
64 * @code
65 * std::unique_ptr<Base> base_ptr = std::make_unique<Derived>();
66 *
67 * // Check if cast is possible
68 * if (UniquePtrCast<Derived, Base>::IsPossible(base_ptr.get())) {
69 * auto derived_ptr = UniquePtrCast<Derived, Base>::DoCast(std::move(base_ptr));
70 * // base_ptr is now empty, derived_ptr owns the object
71 * }
72 *
73 * // Or use DoCastIfPossible for conditional casting
74 * auto result = UniquePtrCast<Derived, Base>::DoCastIfPossible(base_ptr);
75 * // If successful: result owns the object, base_ptr is empty
76 * // If failed: result is nullptr, base_ptr still owns the object
77 * @endcode
78 *
79 * @see unique_dyn_cast() for the public API
80 * @see SelfType for CRTP pattern
81 */
82template <typename To, typename From, typename Derived = void>
83struct UniquePtrCast : CastIsPossible<To, From *> {
84 using Self = SelfType<Derived, UniquePtrCast<To, From>>;
85 using CastResultType = std::unique_ptr<std::remove_reference_t<typename CastRetty<To, From>::ret_type>>;
86
87 static inline CastResultType DoCast(std::unique_ptr<From> &&f) {
88 CRLB_ZONE_SCOPED;
89 return CastResultType((typename CastResultType::element_type *)f.release());
90 }
91
92 static inline CastResultType CastFailed() {
93 CRLB_ZONE_SCOPED;
94 return CastResultType(nullptr);
95 }
96
97 static inline CastResultType DoCastIfPossible(std::unique_ptr<From> &f) {
98 CRLB_ZONE_SCOPED;
99 if (!Self::IsPossible(f.get())) return CastFailed();
100 return DoCast(std::move(f));
101 }
102};

108/**
109 * @ingroup casting_detail
110 * @brief Casting adapter for optional-returning casts.
111 *
112 * OptionalValueCast enables casts that return std::optional<To>, where a
113 * successful cast produces an optional with a value, and a failed cast produces
114 * an empty optional. This is useful for value types and copy-constructible casts.
115 *
116 * The implementation:
117 * 1. Inherits IsPossible from CastIsPossible<To, From>
118 * 2. Inherits DoCastIfPossible from DefaultDoCastIfPossible (check then cast)
119 * 3. Provides DoCast that constructs To from From
120 * 4. Returns empty optional on failure
121 *
122 * @tparam To The target value type
123 * @tparam From The source type
124 * @tparam Derived Optional derived class for CRTP customization
125 *
126 * ## Static Methods
127 *
128 * - `CastFailed()`: Returns empty std::optional<To>
129 * - `DoCast(const From &f)`: Constructs To from f, returns std::optional<To>
130 * - `DoCastIfPossible(const From &f)`: Inherited - checks then casts
131 *
132 * @note Inherits IsPossible from CastIsPossible<To, From>
133 * @note Inherits DoCastIfPossible from DefaultDoCastIfPossible
134 * @note Requires To to be constructible from From
135 * @note Uses CRTP pattern via SelfType for customization
136 *
137 * Example:
138 * @code
139 * // Suppose we have types where Derived constructs from Base
140 * Base b = ...;
141 *
142 * if (OptionalValueCast<Derived, Base>::IsPossible(b)) {
143 * auto opt = OptionalValueCast<Derived, Base>::DoCast(b);
144 * // opt contains Derived(b)
145 * }
146 *
147 * // Or use DoCastIfPossible
148 * auto result = OptionalValueCast<Derived, Base>::DoCastIfPossible(b);
149 * // result is either std::optional<Derived>{Derived(b)} or std::nullopt
150 * @endcode
151 *
152 * @see DefaultDoCastIfPossible for conditional casting logic
153 * @see SelfType for CRTP pattern
154 */
155template <typename To, typename From, typename Derived = void>
156struct OptionalValueCast
157 : public CastIsPossible<To, From>,
158 public DefaultDoCastIfPossible<std::optional<To>, From, SelfType<Derived, OptionalValueCast<To, From>>> {
159 static inline std::optional<To> CastFailed() {
160 CRLB_ZONE_SCOPED;
161 return std::optional<To>{};
162 }
163
164 static inline std::optional<To> DoCast(const From &f) {
165 CRLB_ZONE_SCOPED;
166 return To(f);
167 }
168};

174/**
175 * @ingroup casting_detail
176 * @brief Casting adapter for converting pointers to values.
177 *
178 * ValueFromPointerCast enables casting from pointer types (From*) to value
179 * types (To), where To is constructible from From*. This is useful for wrapper
180 * types or smart pointer-like classes that can be constructed from raw pointers.
181 *
182 * The implementation:
183 * 1. Inherits IsPossible from CastIsPossible<To, From*> (pointer type checking)
184 * 2. Inherits CastFailed from NullableValueCastFailed<To> (returns To(nullptr))
185 * 3. Inherits DoCastIfPossible from DefaultDoCastIfPossible (check then cast)
186 * 4. Provides DoCast that constructs To from From*
187 *
188 * @tparam To The target value type (must be constructible from From*)
189 * @tparam From The pointed-to type
190 * @tparam Derived Optional derived class for CRTP customization
191 *
192 * ## Static Methods
193 *
194 * - `DoCast(From *f)`: Constructs To from f, returns To
195 * - `CastFailed()`: Inherited - returns To(nullptr)
196 * - `DoCastIfPossible(From *f)`: Inherited - checks then casts
197 *
198 * @note Inherits IsPossible from CastIsPossible<To, From*>
199 * @note Inherits CastFailed from NullableValueCastFailed<To>
200 * @note Inherits DoCastIfPossible from DefaultDoCastIfPossible
201 * @note Requires To to be constructible from From*
202 * @note Used for pointer-to-wrapper conversions
203 * @note Uses CRTP pattern via SelfType for customization
204 *
205 * Example:
206 * @code
207 * // Suppose To is a wrapper class constructible from pointers
208 * struct Wrapper {
209 * Base *ptr;
210 * Wrapper(Base *p) : ptr(p) {}
211 * Wrapper(std::nullptr_t) : ptr(nullptr) {}
212 * };
213 *
214 * Derived *d = new Derived();
215 *
216 * if (ValueFromPointerCast<Wrapper, Base>::IsPossible(d)) {
217 * Wrapper w = ValueFromPointerCast<Wrapper, Base>::DoCast(d);
218 * // w wraps the pointer
219 * }
220 *
221 * // Or use DoCastIfPossible
222 * auto result = ValueFromPointerCast<Wrapper, Base>::DoCastIfPossible(d);
223 * // result is either Wrapper(d) or Wrapper(nullptr)
224 * @endcode
225 *
226 * @see NullableValueCastFailed for failure handling
227 * @see DefaultDoCastIfPossible for conditional casting logic
228 * @see SelfType for CRTP pattern
229 */
230template <typename To, typename From, typename Derived = void>
231struct ValueFromPointerCast
232 : public CastIsPossible<To, From *>,
233 public NullableValueCastFailed<To>,
234 public DefaultDoCastIfPossible<To, From *, SelfType<Derived, ValueFromPointerCast<To, From>>> {
235 static inline To DoCast(From *f) {
236 CRLB_ZONE_SCOPED;
237 return To(f);
238 }
239};
