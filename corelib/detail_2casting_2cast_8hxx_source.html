28/**
29 * @ingroup casting_detail
30 * @brief Base template for calculating cast return types.
31 *
32 * CastRettyImpl determines the appropriate return type for a cast based on the
33 * source type's qualifiers (const, pointer, reference). It preserves const-correctness
34 * and reference/pointer semantics in the result type.
35 *
36 * The base template handles non-const reference types, returning To&.
37 * Specializations handle:
38 * - const-qualified types (preserves const)
39 * - Pointer types (returns pointers, preserves const)
40 * - std::unique_ptr (returns std::unique_ptr with correct const)
41 *
42 * @tparam To The target type
43 * @tparam From The source type (may include qualifiers)
44 *
45 * ## Member Types
46 *
47 * - `ret_type`: The calculated return type for the cast
48 *
49 * @note Base template: From → To&
50 * @note Used internally by CastRetty via CastRettyWrap
51 *
52 * @see CastRetty for the main return type calculator
53 * @see CastRettyWrap for type simplification integration
54 */
55template <class To, class From>
56struct CastRettyImpl {
57 using ret_type = To &;
58};
59
60/**
61 * @ingroup casting_detail
62 * @brief Specialization for const-qualified types.
63 *
64 * Preserves const qualification in the return type.
65 *
66 * @tparam To The target type
67 * @tparam From The const-qualified source type
68 *
69 * @note const From → const To&
70 */
71template <class To, class From>
72struct CastRettyImpl<To, const From> {
73 using ret_type = const To &;
74};
75
76/**
77 * @ingroup casting_detail
78 * @brief Specialization for pointer types.
79 *
80 * Returns a pointer to the target type.
81 *
82 * @tparam To The target type
83 * @tparam From The pointed-to type
84 *
85 * @note From* → To*
86 */
87template <class To, class From>
88struct CastRettyImpl<To, From *> {
89 using ret_type = To *;
90};

92/**
93 * @ingroup casting_detail
94 * @brief Specialization for pointer-to-const types.
95 *
96 * Returns a pointer to const target type, preserving const on the pointee.
97 *
98 * @tparam To The target type
99 * @tparam From The const-qualified pointed-to type
100 *
101 * @note const From* → const To*
102 */
103template <class To, class From>
104struct CastRettyImpl<To, const From *> {
105 using ret_type = const To *;
106};

108/**
109 * @ingroup casting_detail
110 * @brief Specialization for const-pointer-to-const types.
111 *
112 * Returns a pointer to const target type (top-level const on pointer is dropped).
113 *
114 * @tparam To The target type
115 * @tparam From The const-qualified pointed-to type
116 *
117 * @note const From* const → const To*
118 */
119template <class To, class From>
120struct CastRettyImpl<To, const From *const> {
121 using ret_type = const To *;
122};

124/**
125 * @ingroup casting_detail
126 * @brief Specialization for std::unique_ptr types.
127 *
128 * Returns a std::unique_ptr to the target type, preserving const-correctness
129 * from the pointee. Uses CastRettyImpl<To, From*> to calculate the pointer
130 * type, then wraps it in unique_ptr.
131 *
132 * @tparam To The target type
133 * @tparam From The type contained in the unique_ptr
134 *
135 * @note std::unique_ptr<From> → std::unique_ptr<To>
136 * @note std::unique_ptr<const From> → std::unique_ptr<const To>
137 */
138template <class To, class From>
139struct CastRettyImpl<To, std::unique_ptr<From>> {
140 private:
141 using PointerType = typename CastRettyImpl<To, From *>::ret_type;
142 using ResultType = std::remove_pointer_t<PointerType>;
143
144 public:
145 using ret_type = std::unique_ptr<ResultType>;
146};

148/**
149 * @ingroup casting_detail
150 * @brief Recursive wrapper for return type calculation with type simplification.
151 *
152 * CastRettyWrap integrates SimplifyType's recursive unwrapping with return type
153 * calculation. It recursively simplifies the From type until reaching a "simple"
154 * type, then delegates to CastRettyImpl for the actual return type calculation.
155 *
156 * The base template provides the recursive case, continuing to simplify.
157 * The specialization (when From == SimpleFrom) is the base case, delegating
158 * to CastRettyImpl.
159 *
160 * @tparam To The target type
161 * @tparam From The original source type (potentially with qualifiers)
162 * @tparam SimpleFrom The simplified version of From
163 *
164 * ## Member Types
165 *
166 * - `ret_type`: The calculated return type after simplification
167 *
168 * @note Recursive template - continues until From == SimpleFrom
169 *
170 * @see CastRettyImpl for the base return type calculation
171 * @see CastRetty for the main entry point
172 */
173template <class To, class From, class SimpleFrom>
174struct CastRettyWrap {
175 using ret_type = typename CastRetty<To, SimpleFrom>::ret_type;
176};
177
178/**
179 * @ingroup casting_detail
180 * @brief Specialization for fully simplified types (base case).
181 *
182 * When From and SimpleFrom are the same, simplification is complete.
183 * Delegate to CastRettyImpl for the actual return type calculation.
184 *
185 * @tparam To The target type
186 * @tparam FromTy The fully simplified source type
187 *
188 * ## Member Types
189 *
190 * - `ret_type`: Delegates to CastRettyImpl<To, FromTy>::ret_type
191 */
192template <class To, class FromTy>
193struct CastRettyWrap<To, FromTy, FromTy> {
194 using ret_type = typename CastRettyImpl<To, FromTy>::ret_type;
195};

197/**
198 * @ingroup casting_detail
199 * @brief Main return type calculator with type simplification.
200 *
201 * CastRetty is the primary interface for determining the return type of a cast
202 * operation. It integrates type simplification (via SimplifyType) with return
203 * type calculation (via CastRettyImpl).
204 *
205 * The process:
206 * 1. Simplify From using SimplifyType
207 * 2. Recursively apply simplification via CastRettyWrap
208 * 3. Once simplified, calculate return type via CastRettyImpl
209 *
210 * @tparam To The target type
211 * @tparam From The source type (may have qualifiers, wrappers, etc.)
212 *
213 * ## Member Types
214 *
215 * - `ret_type`: The calculated return type for cast<To>(From)
216 *
217 * @note Used throughout the casting infrastructure to determine return types
218 * @note Preserves const-correctness and reference/pointer semantics
219 *
220 * Example:
221 * @code
222 * // Non-const reference
223 * static_assert(std::is_same_v<CastRetty<Derived, Base>::ret_type, Derived&>);
224 *
225 * // Const reference
226 * static_assert(std::is_same_v<CastRetty<Derived, const Base>::ret_type, const Derived&>);
227 *
228 * // Pointer
229 * static_assert(std::is_same_v<CastRetty<Derived, Base*>::ret_type, Derived*>);
230 *
231 * // Const pointer
232 * static_assert(std::is_same_v<CastRetty<Derived, const Base*>::ret_type, const Derived*>);
233 *
234 * // Unique ptr
235 * static_assert(std::is_same_v<
236 * CastRetty<Derived, std::unique_ptr<Base>>::ret_type,
237 * std::unique_ptr<Derived>>);
238 * @endcode
239 *
240 * @see CastRettyImpl for the core return type calculations
241 * @see CastRettyWrap for type simplification integration
242 */
243template <class To, class From>
244struct CastRetty {
245 using ret_type = typename CastRettyWrap<To, From, typename SimplifyType<From>::SimpleType>::ret_type;
246};

252/**
253 * @ingroup casting_detail
254 * @brief Base template for checking if a cast is possible.
255 *
256 * CastIsPossible determines whether a cast from From to To can succeed by
257 * delegating to the IsaImplWrap infrastructure. This integrates type
258 * simplification with the classof()-based type checking.
259 *
260 * The base template handles general types by simplifying and checking.
261 * Specializations handle:
262 * - std::optional (unwraps and checks the contained value)
263 * - Trivial upcasts (From derives from To, always possible)
264 *
265 * @tparam To The target type
266 * @tparam From The source type
267 * @tparam Enable SFINAE enabler for specializations
268 *
269 * ## Static Methods
270 *
271 * - `IsPossible(const From &f)`: Returns true if cast<To>(f) would succeed
272 *
273 * @note Delegates to IsaImplWrap for the actual type checking
274 * @note Used by CastInfo and conditional casting operations
275 *
276 * @see IsaImplWrap for the type-checking infrastructure
277 */
278template <typename To, typename From, typename Enable>
279struct CastIsPossible {
280 static inline bool IsPossible(const From &f) {
281 CRLB_ZONE_SCOPED;
282 return detail::IsaImplWrap<To, const From, typename detail::SimplifyType<const From>::SimpleType>::Check(f);
283 }
284};

286/**
287 * @ingroup casting_detail
288 * @brief Specialization for std::optional types.
289 *
290 * Checks if the cast is possible for the contained value. Asserts that the
291 * optional has a value before checking.
292 *
293 * @tparam To The target type
294 * @tparam From The type contained in the optional
295 *
296 * ## Static Methods
297 *
298 * - `IsPossible(const std::optional<From> &f)`: Unwraps f and checks if cast
299 * is possible (asserts f.has_value())
300 *
301 * @note Asserts the optional is not empty using debug::Assert
302 */
303template <typename To, typename From>
304struct CastIsPossible<To, std::optional<From>> {
305 static inline bool IsPossible(const std::optional<From> &f) {
306 CRLB_ZONE_SCOPED;
307 debug::Assert(f.has_value(),
308 "CastIsPossible<{}, std::optional<{}>>::IsPossible(const std::optional<{}>) called on a nullopt!",
309 type_name<To>(), type_name<From>(), type_name<From>());
310 return detail::IsaImplWrap<To, const From, typename detail::SimplifyType<const From>::SimpleType>::Check(*f);
311 }
312};

342/**
343 * @ingroup casting_detail
344 * @brief Recursive cast conversion with type simplification.
345 *
346 * CastConvertVal performs the actual cast operation by:
347 * 1. Recursively simplifying the From type using SimplifyType
348 * 2. Once simplified, performing the cast via pointer/reference manipulation
349 *
350 * The base template provides the recursive case. Specializations handle:
351 * - Fully simplified reference types (base case)
352 * - Pointer types (simple pointer cast)
353 *
354 * @tparam To The target type
355 * @tparam From The original source type (potentially with qualifiers)
356 * @tparam SimpleFrom The simplified version of From
357 *
358 * ## Static Methods
359 *
360 * - `Convert(const From &Val)`: Recursively simplifies and casts Val to To
361 *
362 * @note Recursive template - continues until From == SimpleFrom
363 * @note Uses pointer/reference casting for the actual conversion
364 *
365 * @see CastRetty for return type calculation
366 * @see SimplifyType for type simplification
367 */
368template <class To, class From, class SimpleFrom>
369struct CastConvertVal {
370 static typename CastRetty<To, From>::ret_type Convert(const From &Val) {
371 CRLB_ZONE_SCOPED;
372 return CastConvertVal<To, SimpleFrom, typename SimplifyType<SimpleFrom>::SimpleType>::Convert(
373 SimplifyType<From>::GetSimplifiedValue(const_cast<From &>(Val)));
374 }
375};
376
377/**
378 * @ingroup casting_detail
379 * @brief Specialization for fully simplified reference types (base case).
380 *
381 * When From == SimpleFrom, simplification is complete. Perform the cast by
382 * pointer manipulation: take address of Val, cast pointer, then dereference.
383 *
384 * @tparam To The target type
385 * @tparam FromTy The fully simplified source type
386 *
387 * ## Static Methods
388 *
389 * - `Convert(const FromTy &Val)`: Casts &Val to target pointer type, dereferences
390 *
391 * @note Uses pointer casting with const_cast for the conversion
392 * @note Result is dereferenced to return a reference
393 */
394template <class To, class FromTy>
395struct CastConvertVal<To, FromTy, FromTy> {
396 static typename CastRetty<To, FromTy>::ret_type Convert(const FromTy &Val) {
397 CRLB_ZONE_SCOPED;
398 return *(std::remove_reference_t<typename CastRetty<To, FromTy>::ret_type> *)&const_cast<FromTy &>(Val);
399 }
400};
401
402/**
403 * @ingroup casting_detail
404 * @brief Specialization for pointer types.
405 *
406 * For pointer types, the cast is a simple pointer cast (no dereferencing needed).
407 *
408 * @tparam To The target type
409 * @tparam FromTy The pointed-to type
410 *
411 * ## Static Methods
412 *
413 * - `Convert(const FromTy *Val)`: Casts Val to target pointer type
414 *
415 * @note Direct pointer cast, no pointer/reference manipulation needed
416 */
417template <class To, class FromTy>
418struct CastConvertVal<To, FromTy *, FromTy *> {
419 static typename CastRetty<To, FromTy *>::ret_type Convert(const FromTy *Val) {
420 CRLB_ZONE_SCOPED;
421 return (typename CastRetty<To, FromTy *>::ret_type) const_cast<FromTy *>(Val);
422 }
423};

429/**
430 * @ingroup casting_detail
431 * @brief Main orchestrator for cast operations.
432 *
433 * CastInfo is the central template that coordinates all casting operations.
434 * It integrates:
435 * - Return type calculation (CastRetty)
436 * - Type checking (CastIsPossible)
437 * - Cast conversion (CastConvertVal)
438 * - Conditional casting (DoCastIfPossible pattern)
439 *
440 * The base template handles "simple" types (those that don't need further
441 * simplification). Specializations handle:
442 * - Non-simple types (recursive simplification)
443 * - std::unique_ptr (delegates to UniquePtrCast)
444 * - std::optional (delegates to OptionalValueCast)
445 *
446 * @tparam To The target type
447 * @tparam From The source type
448 * @tparam Enable SFINAE enabler for specializations
449 *
450 * ## Member Types
451 *
452 * - `Self`: The CastInfo type itself
453 * - `CastReturnType`: The return type for the cast (from CastRetty)
454 *
455 * ## Static Methods
456 *
457 * - `DoCast(const From &f)`: Performs the cast (assumes success)
458 * - `CastFailed()`: Returns nullptr (for failed casts)
459 * - `DoCastIfPossible(const From &f)`: Checks then casts
460 * - `IsPossible(const From &f)`: Inherited from CastIsPossible
461 *
462 * @note Inherits IsPossible from CastIsPossible
463 * @note Base template for simple types
464 * @note Used by cast(), dyn_cast(), and related public APIs
465 *
466 * Example:
467 * @code
468 * Base *base = ...;
469 *
470 * // Check if cast is possible
471 * if (CastInfo<Derived, Base*>::IsPossible(base)) {
472 * // Perform the cast
473 * Derived *derived = CastInfo<Derived, Base*>::DoCast(base);
474 * }
475 *
476 * // Or use conditional casting
477 * Derived *result = CastInfo<Derived, Base*>::DoCastIfPossible(base);
478 * // result is nullptr if cast failed
479 * @endcode
480 *
481 * @see CastRetty for return type calculation
482 * @see CastIsPossible for possibility checking
483 * @see CastConvertVal for conversion
484 */
485template <typename To, typename From, typename Enable = void>
486struct CastInfo : CastIsPossible<To, From> {
487 using Self = CastInfo<To, From, Enable>;
488
489 using CastReturnType = typename CastRetty<To, From>::ret_type;
490
491 static inline CastReturnType DoCast(const From &f) {
492 CRLB_ZONE_SCOPED;
493 return CastConvertVal<To, From, typename SimplifyType<From>::SimpleType>::Convert(const_cast<From &>(f));
494 }
495
496 static inline CastReturnType CastFailed() { return CastReturnType(nullptr); }
497
498 static inline CastReturnType DoCastIfPossible(const From &f) {
499 CRLB_ZONE_SCOPED;
500 if (!Self::IsPossible(f)) return CastFailed();
501 return DoCast(f);
502 }
503};

505/**
506 * @ingroup casting_detail
507 * @brief Specialization for non-simple types requiring simplification.
508 *
509 * When From is not a "simple" type (IsSimpleType<From>::value is false),
510 * this specialization recursively simplifies the type and forwards all
511 * operations to the simplified CastInfo.
512 *
513 * @tparam To The target type
514 * @tparam From The non-simple source type
515 *
516 * ## Member Types
517 *
518 * - `Self`: The CastInfo type itself
519 * - `SimpleFrom`: The simplified version of From
520 * - `SimplifiedSelf`: CastInfo<To, SimpleFrom>
521 *
522 * ## Static Methods
523 *
524 * - `IsPossible(From &f)`: Simplifies f and forwards to SimplifiedSelf
525 * - `DoCast(From &f)`: Simplifies f and forwards to SimplifiedSelf
526 * - `CastFailed()`: Forwards to SimplifiedSelf
527 * - `DoCastIfPossible(From &f)`: Simplifies f and forwards to SimplifiedSelf
528 *
529 * @note Enabled via SFINAE when !IsSimpleType<From>::value
530 * @note All operations delegate to CastInfo<To, SimpleFrom>
531 */
532template <typename To, typename From>
533struct CastInfo<To, From, std::enable_if_t<!IsSimpleType<From>::value>> {
534 using Self = CastInfo<To, From>;
535 using SimpleFrom = typename SimplifyType<From>::SimpleType;
536 using SimplifiedSelf = CastInfo<To, SimpleFrom>;
537
538 static inline bool IsPossible(From &f) {
539 CRLB_ZONE_SCOPED;
540 return SimplifiedSelf::IsPossible(SimplifyType<From>::GetSimplifiedValue(f));
541 }
542
543 static inline decltype(auto) DoCast(From &f) {
544 CRLB_ZONE_SCOPED;
545 return SimplifiedSelf::DoCast(SimplifyType<From>::GetSimplifiedValue(f));
546 }
547
548 static inline decltype(auto) CastFailed() {
549 CRLB_ZONE_SCOPED;
550 return SimplifiedSelf::CastFailed();
551 }
552
553 static inline decltype(auto) DoCastIfPossible(From &f) {
554 CRLB_ZONE_SCOPED;
555 return SimplifiedSelf::DoCastIfPossible(SimplifyType<From>::GetSimplifiedValue(f));
556 }
557};
