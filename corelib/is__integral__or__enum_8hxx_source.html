11/**
12 * @ingroup type_traits
13 * @brief Type trait to check if a type is an integral type or enumeration.
14 *
15 * This trait checks whether a type (after removing references) is either an
16 * integral type or an enumeration type. It filters out classes (to avoid
17 * matching types with conversion operators), pointers, and floating-point types.
18 *
19 * The check is performed by:
20 * 1. Removing reference qualifiers from T
21 * 2. Ensuring the type is not a class (filters conversion operators)
22 * 3. Ensuring the type is not a pointer
23 * 4. Ensuring the type is not a floating-point type
24 * 5. Checking if the type is an enum OR convertible to unsigned long long
25 *
26 * @tparam T The type to check
27 *
28 * ## Member Constants
29 *
30 * - `value`: true if T is an integral or enumeration type, false otherwise
31 *
32 * ## Helper Variable Template
33 *
34 * `is_integral_or_enum_v<T>` provides a convenient way to access the value:
35 * @code
36 * constexpr bool b = is_integral_or_enum_v<int>; // true
37 * @endcode
38 *
39 * @note References are stripped before checking
40 * @note Classes with conversion operators to integral types are excluded
41 *
42 * Example:
43 * @code
44 * enum Color { Red, Green, Blue };
45 * enum class Status { OK, Error };
46 *
47 * static_assert(is_integral_or_enum_v<int>); // true
48 * static_assert(is_integral_or_enum_v<unsigned long>); // true
49 * static_assert(is_integral_or_enum_v<bool>); // true
50 * static_assert(is_integral_or_enum_v<char>); // true
51 * static_assert(is_integral_or_enum_v<Color>); // true
52 * static_assert(is_integral_or_enum_v<Status>); // true
53 * static_assert(is_integral_or_enum_v<int&>); // true (reference stripped)
54 *
55 * static_assert(!is_integral_or_enum_v<float>); // false (floating-point)
56 * static_assert(!is_integral_or_enum_v<double>); // false (floating-point)
57 * static_assert(!is_integral_or_enum_v<int*>); // false (pointer)
58 * static_assert(!is_integral_or_enum_v<std::string>); // false (class)
59 * @endcode
60 */
61template <typename T>
62class is_integral_or_enum {
63 using UnderlyingT = std::remove_reference_t<T>;
64
65 public:
66 static const bool value = !std::is_class_v<UnderlyingT> && // Filter conversion operators.
67 !std::is_pointer_v<UnderlyingT> && !std::is_floating_point_v<UnderlyingT> &&
68 (std::is_enum_v<UnderlyingT> || std::is_convertible_v<UnderlyingT, unsigned long long>);
69};
70
71/**
72 * @ingroup type_traits
73 * @brief Helper variable template for is_integral_or_enum.
74 *
75 * Provides convenient access to is_integral_or_enum<T>::value following
76 * the C++17 standard library convention.
77 *
78 * @tparam T The type to check
79 *
80 * Example:
81 * @code
82 * if constexpr (is_integral_or_enum_v<T>) {
83 * // T is integral or enum
84 * }
85 * @endcode
86 */
87template <typename T>
88inline constexpr bool is_integral_or_enum_v = is_integral_or_enum<T>::value;
