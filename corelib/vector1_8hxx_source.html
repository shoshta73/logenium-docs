15/**
16 * @ingroup math_vectors
17 * @brief Template-based 1D vector class.
18 *
19 * vec1 represents a one-dimensional vector with a single component accessible
20 * through multiple naming schemes (x/r/s) for different use cases.
21 *
22 * @tparam T The component type (defaults to float)
23 *
24 * @note Most operations are constexpr and can be evaluated at compile time
25 * @note Length() is non-constexpr due to std::abs
26 *
27 * Example:
28 * @code
29 * vec1<float> v{5.0f};
30 * v += vec1<float>{2.0f}; // v.x == 7.0f
31 * float len = v.Length(); // len == 7.0f
32 * vec1<float> unit = v.Normalized(); // unit.x == 1.0f
33 * @endcode
34 */
35template <typename T = float>
36struct vec1 {
37 /**
38 * @brief Union providing multiple naming schemes for the component.
39 *
40 * - x: Spatial coordinate
41 * - r: Color (red) component
42 * - s: Texture coordinate
43 */
44 union {
45 T x;
46 T r;
47 T s;
48 };
49
50 /**
51 * @brief Default constructor, initializes component to zero.
52 */
53 constexpr vec1() : x(T{0}) { CRLB_ZONE_SCOPED; }
54
55 /**
56 * @brief Construct from a single value.
57 * @param val The value for the component
58 */
59 constexpr vec1(T val) : x(val) { CRLB_ZONE_SCOPED; }
60
61 /**
62 * @brief Create a zero vector (0).
63 * @return vec1 with component set to zero
64 */
65 static constexpr vec1 Zero() {
66 CRLB_ZONE_SCOPED;
67 return {T{0}};
68 }
69
70 /**
71 * @brief Create a unit vector (1).
72 * @return vec1 with component set to one
73 */
74 static constexpr vec1 One() {
75 CRLB_ZONE_SCOPED;
76 return {T{1}};
77 }
78
79 constexpr vec1 operator+(const vec1 &other) const {
80 CRLB_ZONE_SCOPED;
81 return {x + other.x};
82 }
83
84 constexpr vec1 operator-(const vec1 &other) const {
85 CRLB_ZONE_SCOPED;
86 return {x - other.x};
87 }
88
89 constexpr vec1 operator*(const vec1 &other) const {
90 CRLB_ZONE_SCOPED;
91 return {x * other.x};
92 }
93
94 constexpr vec1 operator/(const vec1 &other) const {
95 CRLB_ZONE_SCOPED;
96 debug::Assert(other.x != T{0}, "Division by zero");
97 return {x / other.x};
98 }
99
100 constexpr vec1 &operator+=(const vec1 &other) {
101 CRLB_ZONE_SCOPED;
102 x += other.x;
103 return *this;
104 }
105
106 constexpr vec1 &operator-=(const vec1 &other) {
107 CRLB_ZONE_SCOPED;
108 x -= other.x;
109 return *this;
110 }
111
112 constexpr vec1 operator*(T scalar) const {
113 CRLB_ZONE_SCOPED;
114 return {x * scalar};
115 }
116
117 constexpr vec1 operator/(T scalar) const {
118 CRLB_ZONE_SCOPED;
119 debug::Assert(scalar != T{0}, "Division by zero");
120 return {x / scalar};
121 }
122
123 constexpr bool operator==(const vec1 &other) const {
124 CRLB_ZONE_SCOPED;
125 return x == other.x;
126 }
127
128 constexpr bool operator!=(const vec1 &other) const {
129 CRLB_ZONE_SCOPED;
130 return x != other.x;
131 }
132
133 /**
134 * @brief Calculate the squared length of the vector.
135 * @return The squared length (x * x)
136 * @note Faster than Length() as it avoids the square root operation
137 */
138 constexpr T LengthSquared() const { return x * x; }
139
140 /**
141 * @brief Calculate the length (magnitude) of the vector.
142 * @return The absolute value of the component
143 * @note Uses std::abs for safety with floating-point types
144 * @note Non-constexpr due to std::abs
145 */
146 T Length() const {
147 CRLB_ZONE_SCOPED;
148 // abs is safer for floating point than sqrt(x*x) for 1D
149 return std::abs(x);
150 }
151
152 /**
153 * @brief Normalize the vector in-place to unit length.
154 * @return Reference to this vector after normalization
155 * @note For 1D vectors, this effectively computes the sign (-1, 0, or 1)
156 * @note Zero vectors remain zero
157 */
158 vec1 &Normalize() {
159 CRLB_ZONE_SCOPED;
160 if (x != 0) {
161 x = (x > 0) ? T{1} : T{-1}; // Signum essentially
162 } else {
163 x = T{0};
164 }
165
166 return *this;
167 }
168
169 /**
170 * @brief Get a normalized copy of this vector.
171 * @return A new vec1 normalized to unit length
172 * @note For 1D vectors, this effectively computes the sign (-1, 0, or 1)
173 * @note Zero vectors return zero
174 * @note This method is constexpr
175 */
176 constexpr vec1 Normalized() const {
177 CRLB_ZONE_SCOPED;
178 if (x != 0) {
179 return vec1{(x > 0) ? T{1} : T{-1}};
180 } else {
181 return vec1{T{0}};
182 }
183 }
184
185 /**
186 * @brief Calculate the dot product of two vectors.
187 * @param a The first vector
188 * @param b The second vector
189 * @return The dot product (a.x * b.x)
190 */
191 static constexpr T Dot(const vec1 &a, const vec1 &b) {
192 CRLB_ZONE_SCOPED;
193 return a.x * b.x;
194 }
195};
