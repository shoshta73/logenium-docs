15/**
16 * @ingroup utility
17 * @brief RAII wrapper for automatic resource management with custom releasers.
18 *
19 * AutoRelease manages resources that require custom cleanup logic, automatically
20 * calling the releaser function when the object is destroyed or reset. This is
21 * useful for C-style handles, file descriptors, or other resources that don't
22 * follow RAII patterns natively.
23 *
24 * @tparam T The type of the resource to manage
25 * @tparam InvalidValue The sentinel value indicating an invalid/uninitialized resource
26 * (default: value-initialized T{})
27 *
28 * @note This class is move-only (not copyable)
29 * @note The releaser is only called if the resource value != InvalidValue
30 * @note Move operations are noexcept and leave the moved-from object in a valid state
31 *
32 * Example:
33 * @code
34 * // Managing a file descriptor
35 * AutoRelease<int, -1> fd(open("file.txt", O_RDONLY), [](int fd) { close(fd); });
36 * if (fd) {
37 * // Use fd as an int
38 * read(fd, buffer, size);
39 * }
40 * // fd is automatically closed when it goes out of scope
41 *
42 * // Managing a Windows HANDLE
43 * AutoRelease<HANDLE, nullptr> handle(
44 * CreateFile(...),
45 * [](HANDLE h) { CloseHandle(h); }
46 * );
47 *
48 * // Managing a custom resource
49 * AutoRelease<char*, nullptr> str(
50 * strdup("hello"),
51 * [](char* s) { free(s); }
52 * );
53 * @endcode
54 */
55template <typename T, T InvalidValue = {}>
56class AutoRelease {
57 public:
58 /**
59 * @brief Default constructor, creates an AutoRelease with an invalid resource.
60 */
61 AutoRelease() : AutoRelease(InvalidValue, nullptr) {};
62
63 /**
64 * @brief Construct an AutoRelease with a resource and releaser function.
65 *
66 * @param obj The resource to manage
67 * @param releaser The function to call when releasing the resource
68 */
69 AutoRelease(T obj, std::function<void(T)> releaser) : obj(obj), releaser(releaser) {}
70
71 /**
72 * @brief Destructor that releases the resource if valid.
73 *
74 * Calls the releaser function if obj != InvalidValue and releaser is set.
75 */
76 ~AutoRelease() {
77 CRLB_ZONE_SCOPED;
78 if ((obj != InvalidValue) && releaser) {
79 releaser(obj);
80 }
81 }
82
83 /**
84 * @brief Deleted copy constructor (move-only semantics).
85 */
86 AutoRelease(const AutoRelease &) = delete;
87
88 /**
89 * @brief Deleted copy assignment operator (move-only semantics).
90 */
91 AutoRelease &operator=(const AutoRelease &) = delete;
92
93 /**
94 * @brief Move constructor that transfers ownership from another AutoRelease.
95 *
96 * @param other The AutoRelease to move from (left in valid but unspecified state)
97 */
98 AutoRelease(AutoRelease &&other) noexcept : AutoRelease() {
99 CRLB_ZONE_SCOPED;
100 Swap(other);
101 }
102
103 /**
104 * @brief Move assignment operator that transfers ownership from another AutoRelease.
105 *
106 * Releases the current resource (if valid) before acquiring the new one.
107 *
108 * @param other The AutoRelease to move from (left in valid but unspecified state)
109 * @return Reference to this object
110 */
111 auto operator=(AutoRelease &&other) noexcept -> AutoRelease & {
112 CRLB_ZONE_SCOPED;
113 AutoRelease new_obj{std::move(other)};
114 Swap(new_obj);
115
116 return *this;
117 }
118
119 /**
120 * @brief Swap the contents of this AutoRelease with another.
121 *
122 * @param other The AutoRelease to swap with
123 */
124 void Swap(AutoRelease &other) noexcept {
125 CRLB_ZONE_SCOPED;
126 std::ranges::swap(obj, other.obj);
127 std::ranges::swap(releaser, other.releaser);
128 }
129
130 /**
131 * @brief Release the current resource and acquire a new one.
132 *
133 * If the current resource is valid, it is released before setting the new value.
134 *
135 * @param obj The new resource value to manage
136 */
137 void Reset(T obj) {
138 CRLB_ZONE_SCOPED;
139 if ((this->obj != InvalidValue) && releaser) {
140 releaser(this->obj);
141 }
142 this->obj = obj;
143 }
144
145 /**
146 * @brief Get the managed resource value.
147 *
148 * @return The current resource value
149 */
150 T Get() const { return obj; }
151
152 /**
153 * @brief Implicit conversion to the resource type.
154 *
155 * Allows using AutoRelease directly where T is expected.
156 *
157 * @return The current resource value
158 */
159 operator T() const { return obj; }
160
161 /**
162 * @brief Check if the resource is valid (not equal to InvalidValue).
163 *
164 * @return true if the resource is valid, false otherwise
165 */
166 explicit operator bool() const { return obj != InvalidValue; }
167
168 /**
169 * @brief Get the address of the managed resource.
170 *
171 * Useful for output parameters in C APIs.
172 *
173 * @return Pointer to the managed resource
174 *
175 * Example:
176 * @code
177 * AutoRelease<HANDLE, nullptr> handle;
178 * CreateFileHandle(&handle); // Uses operator&
179 * @endcode
180 */
181 T *operator&() noexcept { return std::addressof(obj); }
182
183 /**
184 * @brief Dereference operator (returns the resource value).
185 *
186 * @return The current resource value
187 */
188 T operator*() const noexcept { return obj; }
189
190 private:
191 T obj;
192 std::function<void(T)> releaser;
193};
194
195} // namespace corelib
