17/**
18 * @ingroup casting_detail
19 * @brief Provides default CastFailed() implementation for nullable types.
20 *
21 * This helper struct provides a static CastFailed() method that returns
22 * To(nullptr), suitable for nullable types (pointers, smart pointers, etc.).
23 * It's used as a base class or building block for cast operations that need
24 * to return a null value when a cast fails.
25 *
26 * @tparam To The target nullable type (must be constructible from nullptr)
27 *
28 * ## Static Methods
29 *
30 * - `CastFailed()`: Returns To(nullptr) to indicate cast failure
31 *
32 * @note Only suitable for nullable types (pointers, std::unique_ptr, etc.)
33 * @note Used internally by dyn_cast_or_null() and similar operations
34 *
35 * Example:
36 * @code
37 * // For pointer types
38 * int *failed_ptr = NullableValueCastFailed<int*>::CastFailed();
39 * // failed_ptr is nullptr
40 *
41 * // For smart pointers
42 * auto failed_unique = NullableValueCastFailed<std::unique_ptr<int>>::CastFailed();
43 * // failed_unique is an empty unique_ptr
44 * @endcode
45 *
46 * @see DefaultDoCastIfPossible for conditional casting logic
47 */
48template <typename To>
49struct NullableValueCastFailed {
50 static To CastFailed() {
51 CRLB_ZONE_SCOPED;
52 return To(nullptr);
53 }
54};

56/**
57 * @ingroup casting_detail
58 * @brief CRTP base providing default DoCastIfPossible() implementation.
59 *
60 * This template implements the conditional casting pattern using CRTP
61 * (Curiously Recurring Template Pattern). It provides a DoCastIfPossible()
62 * method that checks if a cast is possible before attempting it, returning
63 * a failure value if the cast would fail.
64 *
65 * The derived class must provide:
66 * - `IsPossible(From f)`: Returns true if the cast can succeed
67 * - `DoCast(From f)`: Performs the actual cast (assumes success)
68 * - `CastFailed()`: Returns the value to use when cast fails
69 *
70 * This pattern is used throughout the casting infrastructure to implement
71 * dyn_cast() and similar operations that need to check type compatibility
72 * before casting.
73 *
74 * @tparam To The target type of the cast
75 * @tparam From The source type of the cast
76 * @tparam Derived The derived class implementing the actual cast logic (CRTP)
77 *
78 * ## Static Methods
79 *
80 * - `DoCastIfPossible(From f)`: Conditionally casts f to To, calling
81 * Derived::IsPossible() to check, Derived::DoCast() to perform, and
82 * Derived::CastFailed() if the cast would fail
83 *
84 * @note Uses CRTP pattern - Derived must inherit from this class
85 * @note Provides the implementation for dyn_cast() operations
86 *
87 * Example:
88 * @code
89 * template <typename To, typename From>
90 * struct MyCastImpl : DefaultDoCastIfPossible<To, From, MyCastImpl<To, From>> {
91 * static bool IsPossible(From f) {
92 * // Check if cast is possible
93 * return; // type check logic
94 * }
95 *
96 * static To DoCast(From f) {
97 * // Perform the cast
98 * return static_cast<To>(f);
99 * }
100 *
101 * static To CastFailed() {
102 * // Return failure value
103 * return To(nullptr);
104 * }
105 * };
106 *
107 * // Now MyCastImpl has DoCastIfPossible() automatically
108 * auto result = MyCastImpl<Derived*, Base*>::DoCastIfPossible(base_ptr);
109 * @endcode
110 *
111 * @see NullableValueCastFailed for a common CastFailed() implementation
112 * @see SelfType for CRTP pattern support
113 */
114template <typename To, typename From, typename Derived>
115struct DefaultDoCastIfPossible {
116 static To DoCastIfPossible(From f) {
117 CRLB_ZONE_SCOPED;
118 if (!Derived::IsPossible(f)) return Derived::CastFailed();
119 return Derived::DoCast(f);
120 }
121};

127/**
128 * @ingroup casting_detail
129 * @brief Forwards reference-based casts to pointer-based cast implementations.
130 *
131 * This forwarding strategy enables reference casts to be implemented in terms
132 * of pointer casts. It takes a reference, converts it to a pointer, forwards
133 * to the pointer-based cast implementation, then dereferences the result.
134 *
135 * This is a key optimization in the casting infrastructure - reference casts
136 * don't need separate implementations, they can reuse the pointer cast logic.
137 *
138 * @tparam To The target type (typically a reference type)
139 * @tparam From The source reference type
140 * @tparam ForwardTo The pointer-based cast implementation to forward to
141 *
142 * ## Static Methods
143 *
144 * - `IsPossible(const From &f)`: Checks if cast is possible by taking address
145 * of f and calling ForwardTo::IsPossible(&f)
146 * - `DoCast(const From &f)`: Performs cast by taking address of f, calling
147 * ForwardTo::DoCast(&f), then dereferencing the result
148 *
149 * @note Assumes ForwardTo operates on pointer types
150 * @note Result is dereferenced, converting pointer back to reference
151 * @note Uses decltype(auto) to preserve reference types and const-correctness
152 *
153 * Example:
154 * @code
155 * // Suppose we have a pointer cast implementation
156 * struct MyPointerCast {
157 * static bool IsPossible(Base *ptr) { return isa<Derived>(ptr); }
158 * static Derived *DoCast(Base *ptr) { return static_cast<Derived*>(ptr); }
159 * };
160 *
161 * // ForwardToPointerCast enables reference casts
162 * using MyRefCast = ForwardToPointerCast<Derived&, Base&, MyPointerCast>;
163 *
164 * Base &ref = ...;
165 * if (MyRefCast::IsPossible(ref)) {
166 * Derived &d = MyRefCast::DoCast(ref); // Forwarded to pointer cast, then dereferenced
167 * }
168 * @endcode
169 *
170 * @see ConstStrippingForwardingCast for const-aware forwarding
171 */
172template <typename To, typename From, typename ForwardTo>
173struct ForwardToPointerCast {
174 static inline bool IsPossible(const From &f) {
175 CRLB_ZONE_SCOPED;
176 return ForwardTo::IsPossible(&f);
177 }
178
179 static inline decltype(auto) DoCast(const From &f) {
180 CRLB_ZONE_SCOPED;
181 return *ForwardTo::DoCast(&f);
182 }
183};

185/**
186 * @ingroup casting_detail
187 * @brief Forwards const-qualified casts to non-const cast implementations.
188 *
189 * This forwarding strategy handles const-correctness by stripping const
190 * qualifiers from the source type, forwarding to a non-const cast implementation,
191 * then allowing the result to inherit appropriate const-ness. This enables
192 * const-qualified casts to reuse non-const implementations.
193 *
194 * The strategy works for both pointers and references:
195 * - For `const T*`: strips to `T*`, forwards, result gets const back via return type
196 * - For `const T&`: strips to `T&`, forwards, result gets const back via return type
197 *
198 * This is essential for implementing the const-correctness guarantees of the
199 * casting system - casting from `const Base*` should produce `const Derived*`,
200 * but the implementation can reuse the non-const logic.
201 *
202 * @tparam To The target type
203 * @tparam From The source type (may be const-qualified pointer or reference)
204 * @tparam ForwardTo The non-const cast implementation to forward to
205 *
206 * ## Member Types
207 *
208 * - `DecayedFrom`: From with all cv-qualifiers and pointer removed
209 * - `NonConstFrom`: From reconstructed without const (T* for pointers, T& for refs)
210 *
211 * ## Static Methods
212 *
213 * - `IsPossible(const From &f)`: Strips const from f and calls ForwardTo::IsPossible()
214 * - `CastFailed()`: Forwards to ForwardTo::CastFailed()
215 * - `DoCast(const From &f)`: Strips const from f and calls ForwardTo::DoCast()
216 * - `DoCastIfPossible(const From &f)`: Strips const from f and calls
217 * ForwardTo::DoCastIfPossible()
218 *
219 * @note Uses const_cast to safely remove const before forwarding
220 * @note Preserves pointer vs reference distinction
221 * @note Return type preserves const-correctness automatically
222 *
223 * Example:
224 * @code
225 * // Suppose we have a non-const cast implementation
226 * struct NonConstCastImpl {
227 * static bool IsPossible(Base *ptr) { return isa<Derived>(ptr); }
228 * static Derived *DoCast(Base *ptr) { return static_cast<Derived*>(ptr); }
229 * static Derived *CastFailed() { return nullptr; }
230 * };
231 *
232 * // ConstStrippingForwardingCast enables const casts
233 * using ConstCastImpl = ConstStrippingForwardingCast<
234 * const Derived*, const Base*, NonConstCastImpl>;
235 *
236 * const Base *const_base = ...;
237 * if (ConstCastImpl::IsPossible(const_base)) {
238 * // Const is stripped, forwarded, then preserved in return type
239 * const Derived *const_derived = ConstCastImpl::DoCast(const_base);
240 * }
241 * @endcode
242 *
243 * @see ForwardToPointerCast for reference-to-pointer forwarding
244 */
245template <typename To, typename From, typename ForwardTo>
246struct ConstStrippingForwardingCast {
247 using DecayedFrom = std::remove_cv_t<std::remove_pointer_t<From>>;
248 using NonConstFrom = std::conditional_t<std::is_pointer_v<From>, DecayedFrom *, DecayedFrom &>;
249
250 static inline bool IsPossible(const From &f) {
251 CRLB_ZONE_SCOPED;
252 return ForwardTo::IsPossible(const_cast<NonConstFrom>(f));
253 }
254
255 static inline decltype(auto) CastFailed() {
256 CRLB_ZONE_SCOPED;
257 return ForwardTo::CastFailed();
258 }
259
260 static inline decltype(auto) DoCast(const From &f) {
261 CRLB_ZONE_SCOPED;
262 return ForwardTo::DoCast(const_cast<NonConstFrom>(f));
263 }
264
265 static inline decltype(auto) DoCastIfPossible(const From &f) {
266 CRLB_ZONE_SCOPED;
267 return ForwardTo::DoCastIfPossible(const_cast<NonConstFrom>(f));
268 }
269};
