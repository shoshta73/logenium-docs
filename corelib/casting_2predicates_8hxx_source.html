11/**
12 * @ingroup casting
13 * @brief Predicate functor for variadic type checking with algorithms.
14 *
15 * IsaPred is a constexpr variable template that provides a predicate functor
16 * for use with standard algorithms (find_if, all_of, count_if, etc.) and ranges.
17 * It wraps isa<Types...>() to check if a value is an instance of any of the
18 * specified types.
19 *
20 * The predicate supports variadic type checking with OR semantics - returns true
21 * if the value matches ANY of the specified types. This makes it ideal for
22 * filtering, searching, and counting operations in containers.
23 *
24 * @tparam Types The target types to check for (variadic)
25 *
26 * @note Asserts if value is null (use IsaAndPresentPred for null-safe checking)
27 * @note Requires Types to implement classof() for RTTI-free type checking
28 * @note Works with pointers, references, and unique_ptr
29 *
30 * Example:
31 * @code
32 * std::vector<Base*> bases = {new Derived1(), new Derived2(), new Base()};
33 *
34 * // Find first Derived1
35 * auto it = std::find_if(bases.begin(), bases.end(), IsaPred<Derived1>);
36 *
37 * // Count Derived1 or Derived2 instances
38 * auto count = std::count_if(bases.begin(), bases.end(),
39 * IsaPred<Derived1, Derived2>);
40 *
41 * // Check if all are Derived types
42 * bool all_derived = std::all_of(bases.begin(), bases.end(),
43 * IsaPred<Derived1, Derived2>);
44 *
45 * // C++20 ranges
46 * auto derived_view = bases | std::views::filter(IsaPred<Derived1>);
47 *
48 * // Lambda-style usage
49 * auto pred = IsaPred<Derived>;
50 * if (pred(base_ptr)) {
51 * // base_ptr is a Derived
52 * }
53 * @endcode
54 *
55 * @see isa() for the underlying type-checking function
56 * @see IsaAndPresentPred for null-safe version
57 */
58template <typename... Types>
59inline constexpr detail::IsaCheckPredicate<Types...> IsaPred{};

61/**
62 * @ingroup casting
63 * @brief Null-safe predicate functor for type checking with algorithms.
64 *
65 * IsaAndPresentPred is a constexpr variable template that provides a null-safe
66 * predicate functor for use with standard algorithms and ranges. It wraps
67 * isa_and_present<Types...>() to check if a value is present (not null/empty)
68 * AND is an instance of any of the specified types.
69 *
70 * Unlike IsaPred, this predicate never asserts - it safely handles null pointers
71 * and empty optionals by returning false. This makes it ideal for filtering
72 * containers that may contain null values.
73 *
74 * @tparam Types The target types to check for (variadic)
75 *
76 * @note No assertions - safe for null/empty values
77 * @note Short-circuits: returns false immediately if value is not present
78 * @note Supports variadic type checking with OR semantics
79 *
80 * Example:
81 * @code
82 * std::vector<Base*> bases = {new Derived(), nullptr, new Base()};
83 *
84 * // Filter to non-null Derived instances
85 * std::vector<Derived*> deriveds;
86 * std::copy_if(bases.begin(), bases.end(),
87 * std::back_inserter(deriveds),
88 * IsaAndPresentPred<Derived>);
89 * // deriveds = {Derived*} (skips nullptr and Base)
90 *
91 * // Count non-null Derived1 or Derived2
92 * auto count = std::count_if(bases.begin(), bases.end(),
93 * IsaAndPresentPred<Derived1, Derived2>);
94 *
95 * // Works with optional
96 * std::vector<std::optional<Base>> opts = {...};
97 * auto has_derived = std::any_of(opts.begin(), opts.end(),
98 * IsaAndPresentPred<Derived>);
99 *
100 * // C++20 ranges - filter to non-null Derived
101 * auto valid_derived = bases
102 * | std::views::filter(IsaAndPresentPred<Derived>)
103 * | std::views::transform([](auto *p) { return cast<Derived>(p); });
104 * @endcode
105 *
106 * @see isa_and_present() for the underlying function
107 * @see IsaPred for version that asserts on null
108 */
109template <typename... Types>
110inline constexpr detail::IsaAndPresentCheckPredicate<Types...> IsaAndPresentPred{};

112/**
113 * @ingroup casting
114 * @brief Functor for static_cast operations with algorithms.
115 *
116 * StaticCastTo is a constexpr variable template that provides a functor
117 * wrapping static_cast<To> for use with algorithms like std::transform.
118 * Uses perfect forwarding to preserve value categories.
119 *
120 * This is useful for simple type conversions where you need function object
121 * syntax, such as transforming containers or composing with ranges.
122 *
123 * @tparam To The target type for the cast
124 *
125 * @note No type checking - performs unchecked static_cast
126 * @note Preserves value categories via perfect forwarding
127 * @note Use CastTo for checked casting with assertions
128 *
129 * Example:
130 * @code
131 * // Numeric conversions
132 * std::vector<double> doubles = {1.5, 2.5, 3.5};
133 * std::vector<int> ints;
134 * std::transform(doubles.begin(), doubles.end(),
135 * std::back_inserter(ints),
136 * StaticCastTo<int>);
137 * // ints = {1, 2, 3}
138 *
139 * // Pointer conversions (use with caution - unchecked!)
140 * std::vector<Derived*> deriveds = {...};
141 * std::vector<Base*> bases;
142 * std::transform(deriveds.begin(), deriveds.end(),
143 * std::back_inserter(bases),
144 * StaticCastTo<Base*>);
145 *
146 * // C++20 ranges composition
147 * auto ints_view = doubles | std::views::transform(StaticCastTo<int>);
148 * @endcode
149 *
150 * @see CastTo for checked casting with assertions
151 * @see DynCastTo for dynamic casting with null on failure
152 */
153template <typename To>
154inline constexpr detail::StaticCastFunc<To> StaticCastTo{};

156/**
157 * @ingroup casting
158 * @brief Functor for checked cast operations with algorithms.
159 *
160 * CastTo is a constexpr variable template that provides a functor wrapping
161 * cast<To> for use with algorithms. Performs type-safe downcast with runtime
162 * type checking and assertion on mismatch.
163 *
164 * Unlike StaticCastTo, this functor performs runtime type checking using the
165 * classof() pattern and asserts if the cast is invalid. Use with algorithms
166 * when you know the types are correct and want to catch errors in debug builds.
167 *
168 * @tparam To The target type for the cast
169 *
170 * @note Asserts on type mismatch (debug::Assert)
171 * @note Requires To to implement classof() for type checking
172 * @note Preserves value categories via perfect forwarding
173 *
174 * Example:
175 * @code
176 * std::vector<Base*> bases = {new Derived(), new Derived()};
177 *
178 * // Transform to derived (asserts if any Base is not Derived)
179 * std::vector<Derived*> deriveds;
180 * std::transform(bases.begin(), bases.end(),
181 * std::back_inserter(deriveds),
182 * CastTo<Derived*>);
183 *
184 * // Use after filtering to ensure all are correct type
185 * auto derived_ptrs = bases
186 * | std::views::filter(IsaPred<Derived>)
187 * | std::views::transform(CastTo<Derived*>);
188 *
189 * // Works with references
190 * std::vector<std::reference_wrapper<Base>> refs = {...};
191 * std::vector<Derived*> ptrs;
192 * std::transform(refs.begin(), refs.end(),
193 * std::back_inserter(ptrs),
194 * [](Base &b) { return &CastTo<Derived&>(b); });
195 * @endcode
196 *
197 * @see cast() for the underlying function
198 * @see DynCastTo for version that returns nullptr on failure
199 * @see CastIfPresentTo for null-safe version
200 */
201template <typename To>
202inline constexpr detail::CastFunc<To> CastTo{};

204/**
205 * @ingroup casting
206 * @brief Functor for null-safe checked cast operations with algorithms.
207 *
208 * CastIfPresentTo is a constexpr variable template that provides a functor
209 * wrapping cast_if_present<To> for use with algorithms. Combines presence
210 * checking with type-safe casting - returns null if value is not present,
211 * asserts if present but wrong type.
212 *
213 * This is useful when transforming containers with nullable types (pointers,
214 * optionals) where you want null-safety but still want assertions on type
215 * mismatches for non-null values.
216 *
217 * @tparam To The target type for the cast
218 *
219 * @note Returns null/empty if value is not present
220 * @note Asserts if value is present but not of type To
221 * @note Preserves value categories via perfect forwarding
222 *
223 * Example:
224 * @code
225 * std::vector<Base*> bases = {new Derived(), nullptr, new Derived()};
226 *
227 * // Transform with null-safety (nullptr stays nullptr)
228 * std::vector<Derived*> deriveds;
229 * std::transform(bases.begin(), bases.end(),
230 * std::back_inserter(deriveds),
231 * CastIfPresentTo<Derived*>);
232 * // deriveds = {Derived*, nullptr, Derived*}
233 *
234 * // Works with optional
235 * std::vector<std::optional<Base>> opts = {Derived(), {}, Derived()};
236 * std::vector<std::optional<Derived>> derived_opts;
237 * std::transform(opts.begin(), opts.end(),
238 * std::back_inserter(derived_opts),
239 * CastIfPresentTo<Derived>);
240 *
241 * // Compose with filtering
242 * auto valid_deriveds = bases
243 * | std::views::transform(CastIfPresentTo<Derived*>)
244 * | std::views::filter([](auto *p) { return p != nullptr; });
245 * @endcode
246 *
247 * @see cast_if_present() for the underlying function
248 * @see CastTo for version that asserts on null
249 * @see DynCastIfPresentTo for version that returns null on type mismatch
250 */
251template <typename To>
252inline constexpr detail::CastIfPresentFunc<To> CastIfPresentTo{};

254/**
255 * @ingroup casting
256 * @brief Functor for null-safe dynamic cast operations with algorithms.
257 *
258 * DynCastIfPresentTo is a constexpr variable template that provides a functor
259 * wrapping dyn_cast_if_present<To> for use with algorithms. This is the safest
260 * casting functor - it returns null if the value is not present OR if the type
261 * is wrong, with no assertions.
262 *
263 * Use this when transforming containers where both null values and type
264 * mismatches should be handled gracefully by returning null rather than
265 * asserting or throwing.
266 *
267 * @tparam To The target type for the cast
268 *
269 * @note No assertions - completely safe
270 * @note Returns null if value is not present
271 * @note Returns null if value is present but wrong type
272 * @note Ideal for filtering pipelines that discard failures
273 *
274 * Example:
275 * @code
276 * std::vector<Base*> bases = {new Derived(), nullptr, new OtherType()};
277 *
278 * // Transform with full safety (nullptr and wrong types become nullptr)
279 * std::vector<Derived*> deriveds;
280 * std::transform(bases.begin(), bases.end(),
281 * std::back_inserter(deriveds),
282 * DynCastIfPresentTo<Derived*>);
283 * // deriveds = {Derived*, nullptr, nullptr}
284 *
285 * // Filter out nulls to get only successful casts
286 * std::vector<Derived*> valid_deriveds;
287 * for (auto *d : bases | std::views::transform(DynCastIfPresentTo<Derived*>)
288 * | std::views::filter([](auto *p) { return p; })) {
289 * valid_deriveds.push_back(d);
290 * }
291 *
292 * // Count successful casts
293 * auto count = std::count_if(bases.begin(), bases.end(),
294 * [](Base *b) {
295 * return DynCastIfPresentTo<Derived*>(b) != nullptr;
296 * });
297 * @endcode
298 *
299 * @see dyn_cast_if_present() for the underlying function
300 * @see DynCastTo for version that asserts on null
301 * @see CastIfPresentTo for version that asserts on type mismatch
302 */
303template <typename To>
304inline constexpr detail::DynCastIfPresentFunc<To> DynCastIfPresentTo{};

306/**
307 * @ingroup casting
308 * @brief Functor for dynamic cast operations with algorithms.
309 *
310 * DynCastTo is a constexpr variable template that provides a functor wrapping
311 * dyn_cast<To> for use with algorithms. Performs type-safe downcast with
312 * runtime checking - returns nullptr on type mismatch but asserts if the
313 * value is null.
314 *
315 * Use this when transforming containers where type mismatches should be handled
316 * gracefully (return nullptr) but null values should be caught (assertion).
317 * This is useful when processing containers that should never contain nulls
318 * but may contain mixed types.
319 *
320 * @tparam To The target type for the cast
321 *
322 * @note Asserts if value is null
323 * @note Returns nullptr if value is present but wrong type (no assertion)
324 * @note Requires To to implement classof() for type checking
325 *
326 * Example:
327 * @code
328 * std::vector<Base*> bases = {new Derived(), new Base(), new Derived()};
329 *
330 * // Transform with dynamic cast (wrong types become nullptr)
331 * std::vector<Derived*> deriveds;
332 * std::transform(bases.begin(), bases.end(),
333 * std::back_inserter(deriveds),
334 * DynCastTo<Derived*>);
335 * // deriveds = {Derived*, nullptr, Derived*}
336 *
337 * // Filter to only successful casts
338 * auto valid_deriveds = bases
339 * | std::views::transform(DynCastTo<Derived*>)
340 * | std::views::filter([](auto *p) { return p != nullptr; });
341 *
342 * // Pattern: separate types from mixed container
343 * std::vector<Derived1*> d1s, d2s;
344 * for (Base *b : bases) {
345 * if (auto *d1 = DynCastTo<Derived1*>(b)) d1s.push_back(d1);
346 * if (auto *d2 = DynCastTo<Derived2*>(b)) d2s.push_back(d2);
347 * }
348 * @endcode
349 *
350 * @see dyn_cast() for the underlying function
351 * @see DynCastIfPresentTo for null-safe version
352 * @see CastTo for version that asserts on type mismatch
353 */
354template <typename To>
355inline constexpr detail::DynCastFunc<To> DynCastTo{};
