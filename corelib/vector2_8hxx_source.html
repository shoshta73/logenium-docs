15/**
16 * @ingroup math_vectors
17 * @brief Template-based 2D vector class.
18 *
19 * vec2 represents a two-dimensional vector with components accessible
20 * through multiple naming schemes (x/y, r/g, s/t) for different use cases.
21 *
22 * @tparam T The component type (float, double, int, etc.)
23 *
24 * @note Most operations are constexpr and can be evaluated at compile time
25 * @note Length() and Normalized() are non-constexpr due to std::sqrt
26 *
27 * Example:
28 * @code
29 * vec2<float> a{1.0f, 2.0f};
30 * vec2<float> b{3.0f, 4.0f};
31 * auto sum = a + b; // (4, 6)
32 * float dot = vec2<float>::Dot(a, b); // 11
33 * vec2<float> unit = a.Normalized(); // (0.447, 0.894)
34 * @endcode
35 */
36template <typename T>
37struct vec2 {
38 /**
39 * @brief Union providing multiple naming schemes for the components.
40 *
41 * - x, y: Spatial coordinates
42 * - r, g: Color (red, green) components
43 * - s, t: Texture coordinates
44 */
45 union {
46 struct {
47 T x;
48 T y;
49 };
50 struct {
51 T r;
52 T g;
53 };
54 struct {
55 T s;
56 T t;
57 };
58 };
59
60 /**
61 * @brief Default constructor, initializes all components to zero.
62 */
63 constexpr vec2() : x(T{0}), y(T{0}) { CRLB_ZONE_SCOPED; }
64
65 /**
66 * @brief Construct from a uniform value.
67 * @param val The value for all components
68 */
69 constexpr vec2(T val) : x(val), y(val) { CRLB_ZONE_SCOPED; }
70
71 /**
72 * @brief Construct from individual components.
73 * @param x The x component
74 * @param y The y component
75 */
76 constexpr vec2(T x, T y) : x(x), y(y) { CRLB_ZONE_SCOPED; }
77
78 /**
79 * @brief Create a zero vector (0, 0).
80 * @return vec2 with all components set to zero
81 */
82 static constexpr vec2 Zero() {
83 CRLB_ZONE_SCOPED;
84 return {T{0}, T{0}};
85 }
86
87 /**
88 * @brief Create a unit vector (1, 1).
89 * @return vec2 with all components set to one
90 */
91 static constexpr vec2 One() { return {T{1}, T{1}}; }
92
93 constexpr vec2 operator+(const vec2 &other) const {
94 CRLB_ZONE_SCOPED;
95 return {x + other.x, y + other.y};
96 }
97 constexpr vec2 operator-(const vec2 &other) const {
98 CRLB_ZONE_SCOPED;
99 return {x - other.x, y - other.y};
100 }
101 constexpr vec2 operator*(const vec2 &other) const {
102 CRLB_ZONE_SCOPED;
103 return {x * other.x, y * other.y};
104 }
105 constexpr vec2 operator/(const vec2 &other) const {
106 CRLB_ZONE_SCOPED;
107 debug::Assert(other.x != T{0} && other.y != T{0}, "Division by zero");
108 return {x / other.x, y / other.y};
109 }
110
111 constexpr vec2 &operator+=(const vec2 &other) {
112 CRLB_ZONE_SCOPED;
113 x += other.x;
114 y += other.y;
115 return *this;
116 }
117 constexpr vec2 &operator-=(const vec2 &other) {
118 CRLB_ZONE_SCOPED;
119 x -= other.x;
120 y -= other.y;
121 return *this;
122 }
123
124 constexpr vec2 operator*(T scalar) const {
125 CRLB_ZONE_SCOPED;
126 return {x * scalar, y * scalar};
127 }
128 constexpr vec2 operator/(T scalar) const {
129 CRLB_ZONE_SCOPED;
130 debug::Assert(scalar != T{0}, "Division by zero");
131 return {x / scalar, y / scalar};
132 }
133
134 constexpr bool operator==(const vec2 &other) const {
135 CRLB_ZONE_SCOPED;
136 return x == other.x && y == other.y;
137 }
138 constexpr bool operator!=(const vec2 &other) const {
139 CRLB_ZONE_SCOPED;
140 return x != other.x || y != other.y;
141 }
142
143 /**
144 * @brief Calculate the squared length of the vector.
145 * @return The squared length (x * x + y * y)
146 * @note Faster than Length() as it avoids the square root operation
147 */
148 constexpr T LengthSquared() const {
149 CRLB_ZONE_SCOPED;
150 return x * x + y * y;
151 }
152
153 /**
154 * @brief Calculate the length (magnitude) of the vector.
155 * @return The Euclidean length sqrt(x * x + y * y)
156 * @note Non-constexpr due to std::sqrt
157 */
158 T Length() const {
159 CRLB_ZONE_SCOPED;
160 return std::sqrt(LengthSquared());
161 }
162
163 /**
164 * @brief Normalize the vector in-place to unit length.
165 * @return Reference to this vector after normalization
166 * @note Asserts if the vector has zero length (use debug::Assert)
167 */
168 vec2 &Normalize() {
169 CRLB_ZONE_SCOPED;
170 T len = Length();
171 debug::Assert(len != T{0}, "Cannot normalize a vector with zero length");
172 x /= len;
173 y /= len;
174 return *this;
175 }
176
177 /**
178 * @brief Get a normalized copy of this vector.
179 * @return A new vec2 normalized to unit length
180 * @note Asserts if the vector has zero length (use debug::Assert)
181 * @note This method is constexpr
182 */
183 constexpr vec2 Normalized() const {
184 CRLB_ZONE_SCOPED;
185 T len = Length();
186 debug::Assert(len != T{0}, "Cannot normalize a vector with zero length");
187 return {x / len, y / len};
188 }
189
190 /**
191 * @brief Calculate the dot product of two vectors.
192 * @param a The first vector
193 * @param b The second vector
194 * @return The dot product (a.x * b.x + a.y * b.y)
195 */
196 static constexpr T Dot(const vec2 &a, const vec2 &b) {
197 CRLB_ZONE_SCOPED;
198 return a.x * b.x + a.y * b.y;
199 }
200};
