22/**
23 * @ingroup casting_detail
24 * @brief Base template for RTTI-free type checking using classof() pattern.
25 *
26 * IsaImpl provides the core type-checking mechanism for the casting system.
27 * It delegates to the target type's static classof() method, implementing the
28 * LLVM-style RTTI-free type checking pattern. Types must provide a classof()
29 * method that returns true if a given pointer is of the target type.
30 *
31 * The base template handles the general case by calling To::classof(&Val).
32 * A specialization handles the trivial case where From derives from To,
33 * which is always true and doesn't require classof().
34 *
35 * @tparam To The target type to check for
36 * @tparam From The source type being checked
37 * @tparam Enabler SFINAE enabler for specializations
38 *
39 * ## Static Methods
40 *
41 * - `Check(const From &Val)`: Returns true if Val is of type To (via classof())
42 *
43 * @note Requires To to have a static classof(const From*) method
44 * @note This is the foundation of isa<To>(Val) functionality
45 *
46 * Example:
47 * @code
48 * struct Base {
49 * virtual ~Base() = default;
50 * static bool classof(const Base *) { return true; }
51 * };
52 *
53 * struct Derived : Base {
54 * static bool classof(const Base *b) {
55 * // Check if b is actually a Derived instance
56 * return; //type check logic
57 * }
58 * };
59 *
60 * Derived d;
61 * bool result = IsaImpl<Derived, Base>::Check(d); // Calls Derived::classof(&d)
62 * @endcode
63 *
64 * @see IsaImplConst for const and pointer handling
65 * @see IsaImplWrap for type simplification integration
66 */
67template <typename To, typename From, typename Enabler = void>
68struct IsaImpl {
69 static inline bool Check(const From &Val) {
70 CRLB_ZONE_SCOPED;
71 return To::classof(&Val);
72 }
73};
74
75/**
76 * @ingroup casting_detail
77 * @brief Specialization for when From is derived from To (always true).
78 *
79 * This specialization handles the trivial case where the source type From
80 * is derived from the target type To. In this case, the check always succeeds
81 * without needing to call classof(), since every Derived instance is also a Base.
82 *
83 * @tparam To The base type
84 * @tparam From The derived type (must derive from To)
85 *
86 * ## Static Methods
87 *
88 * - `Check(const From &)`: Always returns true
89 *
90 * @note Enabled via SFINAE when std::is_base_of_v<To, From> is true
91 * @note Optimization - avoids classof() call for trivial upcasts
92 *
93 * Example:
94 * @code
95 * struct Base { };
96 * struct Derived : Base { };
97 *
98 * Derived d;
99 * bool result = IsaImpl<Base, Derived>::Check(d); // Always true (no classof call)
100 * @endcode
101 */
102template <typename To, typename From>
103struct IsaImpl<To, From, std::enable_if_t<std::is_base_of_v<To, From>>> {
104 static inline bool Check(const From &) {
105 CRLB_ZONE_SCOPED;
106 return true;
107 }
108};

114/**
115 * @ingroup casting_detail
116 * @brief Base template for const-correct type checking with pointer support.
117 *
118 * IsaImplConst extends IsaImpl with const-correctness and pointer/unique_ptr
119 * handling. It provides specializations for various const-qualified types and
120 * dereferences pointers before delegating to IsaImpl.
121 *
122 * The base template simply forwards to IsaImpl for non-pointer, non-const types.
123 * Specializations handle:
124 * - const-qualified types
125 * - Pointer types (with null checking)
126 * - std::unique_ptr (with null checking)
127 * - Various combinations of const and pointer
128 *
129 * @tparam To The target type to check for
130 * @tparam From The source type (may include const/pointer qualifiers)
131 *
132 * ## Static Methods
133 *
134 * - `Check(const From &Val)`: Returns true if Val is of type To
135 *
136 * @note Base template forwards to IsaImpl
137 * @note Pointer specializations dereference and assert non-null
138 * @note Const specializations strip const and forward to non-const implementation
139 *
140 * Example:
141 * @code
142 * Derived d;
143 * Derived *ptr = &d;
144 *
145 * // Base template (non-const reference)
146 * bool r1 = IsaImplConst<Base, Derived>::Check(d);
147 *
148 * // Const specialization
149 * bool r2 = IsaImplConst<Base, const Derived>::Check(d);
150 *
151 * // Pointer specialization (dereferences ptr)
152 * bool r3 = IsaImplConst<Base, Derived*>::Check(ptr);
153 * @endcode
154 *
155 * @see IsaImpl for the underlying type-checking mechanism
156 */
157template <typename To, typename From>
158struct IsaImplConst {
159 static inline bool Check(const From &Val) {
160 CRLB_ZONE_SCOPED;
161 return IsaImpl<To, From>::Check(Val);
162 }
163};

183/**
184 * @ingroup casting_detail
185 * @brief Specialization for const std::unique_ptr.
186 *
187 * Dereferences the unique_ptr and recursively checks the pointed-to type.
188 * Asserts that the unique_ptr is not null before dereferencing.
189 *
190 * @tparam To The target type
191 * @tparam From The type contained in the unique_ptr
192 *
193 * @note Asserts Val.get() != nullptr using debug::Assert
194 */
195template <typename To, typename From>
196struct IsaImplConst<To, const std::unique_ptr<From>> {
197 static inline bool Check(const std::unique_ptr<From> &Val) {
198 CRLB_ZONE_SCOPED;
199 debug::Assert(Val.get(), "isa<{}>(const std::unique_ptr<{}> &) called on a null pointer", type_name<To>(),
200 type_name<From>());
201 return IsaImplConst<To, From>::Check(*Val);
202 }
203};

290/**
291 * @ingroup casting_detail
292 * @brief Recursive type simplification wrapper for type checking.
293 *
294 * IsaImplWrap integrates SimplifyType's recursive type unwrapping with the
295 * type-checking infrastructure. It recursively simplifies the From type using
296 * SimplifyType until it reaches a "simple" type (one that SimplifyType doesn't
297 * further simplify), then delegates to IsaImplConst for the actual check.
298 *
299 * This template provides the recursive case, continuing to simplify until
300 * From == SimpleFrom (handled by the specialization).
301 *
302 * @tparam To The target type to check for
303 * @tparam From The original source type (potentially with qualifiers)
304 * @tparam SimpleFrom The simplified version of From (SimplifyType<From>::SimpleType)
305 *
306 * ## Static Methods
307 *
308 * - `Check(const From &Val)`: Recursively simplifies Val and checks type
309 *
310 * @note Recursive template - continues until SimpleFrom is fully simplified
311 * @note Uses SimplifyType::GetSimplifiedValue() to unwrap values
312 *
313 * Example:
314 * @code
315 * // Suppose From = const Base, SimpleFrom = Base
316 * // IsaImplWrap<Derived, const Base, Base>::Check(val)
317 * // Simplifies const Base to Base
318 * // Recursively calls IsaImplWrap<Derived, Base, Base>::Check()
319 * // Reaches specialization, calls IsaImplConst<Derived, Base>::Check()
320 * @endcode
321 *
322 * @see IsaImplWrap<To, FromTy, FromTy> for the base case (fully simplified)
323 * @see SimplifyType for the type simplification mechanism
324 */
325template <typename To, typename From, typename SimpleFrom>
326struct IsaImplWrap {
327 static bool Check(const From &Val) {
328 CRLB_ZONE_SCOPED;
329 return IsaImplWrap<To, SimpleFrom, typename SimplifyType<SimpleFrom>::SimpleType>::Check(
330 SimplifyType<const From>::GetSimplifiedValue(Val));
331 }
332};
333
334/**
335 * @ingroup casting_detail
336 * @brief Specialization for fully simplified types (base case).
337 *
338 * When From and SimpleFrom are the same type, it means SimplifyType has finished
339 * unwrapping and we've reached a "simple" type. At this point, delegate to
340 * IsaImplConst for the actual type check.
341 *
342 * @tparam To The target type to check for
343 * @tparam FromTy The fully simplified source type
344 *
345 * ## Static Methods
346 *
347 * - `Check(const FromTy &Val)`: Delegates to IsaImplConst for type checking
348 *
349 * @note Base case of the recursive simplification
350 * @note FromTy should be a "simple" type with no further unwrapping needed
351 *
352 * Example:
353 * @code
354 * // When recursion reaches a simple type like Base or Base*
355 * IsaImplWrap<Derived, Base, Base>::Check(base_ref)
356 * IsaImplConst<Derived, Base>::Check(base_ref)
357 * IsaImpl<Derived, Base>::Check(base_ref)
358 * Derived::classof(&base_ref)
359 * @endcode
360 *
361 * @see IsaImplConst for const and pointer handling
362 */
363template <typename To, typename FromTy>
364struct IsaImplWrap<To, FromTy, FromTy> {
365 static bool Check(const FromTy &Val) {
366 CRLB_ZONE_SCOPED;
367 return IsaImplConst<To, FromTy>::Check(Val);
368 }
369};
