15/**
16 * @ingroup math_vectors
17 * @brief Template-based 4D vector class.
18 *
19 * Vec4 represents a four-dimensional vector with components accessible
20 * through multiple naming schemes (x/y/z/w, r/g/b/a, s/t/p/q) for different use cases.
21 *
22 * @tparam T The component type (float, double, int, etc.)
23 *
24 * @note Most operations are constexpr and can be evaluated at compile time
25 * @note Length() and Normalized() are non-constexpr due to std::sqrt
26 * @note No Cross() product method (cross product is specific to 3D vectors)
27 *
28 * Example:
29 * @code
30 * Vec4<float> a{1.0f, 2.0f, 3.0f, 4.0f};
31 * Vec4<float> b{5.0f, 6.0f, 7.0f, 8.0f};
32 * auto sum = a + b; // (6, 8, 10, 12)
33 * float dot = Vec4<float>::Dot(a, b); // 70
34 * Vec4<float> unit = a.Normalized(); // (0.182, 0.365, 0.547, 0.730)
35 *
36 * // RGBA color usage
37 * Vec4<float> color{1.0f, 0.5f, 0.25f, 1.0f};
38 * color.a = 0.5f; // Set alpha to 50%
39 * @endcode
40 */
41template <typename T>
42struct vec4 {
43 /**
44 * @brief Union providing multiple naming schemes for the components.
45 *
46 * - x, y, z, w: Spatial/homogeneous coordinates
47 * - r, g, b, a: Color (red, green, blue, alpha) components
48 * - s, t, p, q: Texture coordinates
49 */
50 union {
51 struct {
52 T x;
53 T y;
54 T z;
55 T w;
56 };
57 struct {
58 T r;
59 T g;
60 T b;
61 T a;
62 };
63 struct {
64 T s;
65 T t;
66 T p;
67 T q;
68 };
69 };
70
71 /**
72 * @brief Default constructor, initializes all components to zero.
73 */
74 constexpr vec4() : x(T{0}), y(T{0}), z(T{0}), w(T{0}) { CRLB_ZONE_SCOPED; }
75
76 /**
77 * @brief Construct from a uniform value.
78 * @param val The value for all components
79 */
80 constexpr vec4(T val) : x(val), y(val), z(val), w(val) { CRLB_ZONE_SCOPED; }
81
82 /**
83 * @brief Construct from individual components.
84 * @param x The x component
85 * @param y The y component
86 * @param z The z component
87 * @param w The w component
88 */
89 constexpr vec4(T x, T y, T z, T w) : x(x), y(y), z(z), w(w) { CRLB_ZONE_SCOPED; }
90
91 /**
92 * @brief Create a zero vector (0, 0, 0, 0).
93 * @return Vec4 with all components set to zero
94 */
95 static constexpr vec4 Zero() {
96 CRLB_ZONE_SCOPED;
97 return {T{0}, T{0}, T{0}, T{0}};
98 }
99
100 /**
101 * @brief Create a unit vector (1, 1, 1, 1).
102 * @return Vec4 with all components set to one
103 */
104 static constexpr vec4 One() { return {T{1}, T{1}, T{1}, T{1}}; }
105
106 constexpr vec4 operator+(const vec4 &other) const {
107 CRLB_ZONE_SCOPED;
108 return {x + other.x, y + other.y, z + other.z, w + other.w};
109 }
110 constexpr vec4 operator-(const vec4 &other) const {
111 CRLB_ZONE_SCOPED;
112 return {x - other.x, y - other.y, z - other.z, w - other.w};
113 }
114 constexpr vec4 operator*(const vec4 &other) const {
115 CRLB_ZONE_SCOPED;
116 return {x * other.x, y * other.y, z * other.z, w * other.w};
117 }
118 constexpr vec4 operator/(const vec4 &other) const {
119 CRLB_ZONE_SCOPED;
120 debug::Assert(other.x != T{0} && other.y != T{0} && other.z != T{0} && other.w != T{0}, "Division by zero");
121 return {x / other.x, y / other.y, z / other.z, w / other.w};
122 }
123
124 constexpr vec4 &operator+=(const vec4 &other) {
125 CRLB_ZONE_SCOPED;
126 x += other.x;
127 y += other.y;
128 z += other.z;
129 w += other.w;
130 return *this;
131 }
132 constexpr vec4 &operator-=(const vec4 &other) {
133 CRLB_ZONE_SCOPED;
134 x -= other.x;
135 y -= other.y;
136 z -= other.z;
137 w -= other.w;
138 return *this;
139 }
140
141 constexpr vec4 operator*(T scalar) const {
142 CRLB_ZONE_SCOPED;
143 return {x * scalar, y * scalar, z * scalar, w * scalar};
144 }
145 constexpr vec4 operator/(T scalar) const {
146 CRLB_ZONE_SCOPED;
147 debug::Assert(scalar != T{0}, "Division by zero");
148 return {x / scalar, y / scalar, z / scalar, w / scalar};
149 }
150
151 constexpr bool operator==(const vec4 &other) const {
152 CRLB_ZONE_SCOPED;
153 return x == other.x && y == other.y && z == other.z && w == other.w;
154 }
155 constexpr bool operator!=(const vec4 &other) const {
156 CRLB_ZONE_SCOPED;
157 return x != other.x || y != other.y || z != other.z || w != other.w;
158 }
159
160 /**
161 * @brief Calculate the squared length of the vector.
162 * @return The squared length (x * x + y * y + z * z + w * w)
163 * @note Faster than Length() as it avoids the square root operation
164 */
165 constexpr T LengthSquared() const {
166 CRLB_ZONE_SCOPED;
167 return x * x + y * y + z * z + w * w;
168 }
169
170 /**
171 * @brief Calculate the length (magnitude) of the vector.
172 * @return The Euclidean length sqrt(x * x + y * y + z * z + w * w)
173 * @note Non-constexpr due to std::sqrt
174 */
175 T Length() const {
176 CRLB_ZONE_SCOPED;
177 return std::sqrt(LengthSquared());
178 }
179
180 /**
181 * @brief Normalize the vector in-place to unit length.
182 * @return Reference to this vector after normalization
183 * @note Asserts if the vector has zero length (use debug::Assert)
184 */
185 vec4 &Normalize() {
186 CRLB_ZONE_SCOPED;
187 T len = Length();
188 debug::Assert(len != T{0}, "Cannot normalize a vector with zero length");
189 x /= len;
190 y /= len;
191 z /= len;
192 w /= len;
193 return *this;
194 }
195
196 /**
197 * @brief Get a normalized copy of this vector.
198 * @return A new Vec4 normalized to unit length
199 * @note Asserts if the vector has zero length (use debug::Assert)
200 * @note This method is constexpr
201 */
202 constexpr vec4 Normalized() const {
203 CRLB_ZONE_SCOPED;
204 T len = Length();
205 debug::Assert(len != T{0}, "Cannot normalize a vector with zero length");
206 return {x / len, y / len, z / len, w / len};
207 }
208
209 /**
210 * @brief Calculate the dot product of two vectors.
211 * @param a The first vector
212 * @param b The second vector
213 * @return The dot product (a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w)
214 */
215 static constexpr T Dot(const vec4 &a, const vec4 &b) {
216 CRLB_ZONE_SCOPED;
217 return a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;
218 }
219};
